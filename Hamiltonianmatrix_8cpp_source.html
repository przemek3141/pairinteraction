228 void Hamiltonianmatrix::diagonalize() {
229  if (this->num_basisvectors() > 1) {
230  // diagonalization
231  Eigen::SelfAdjointEigenSolver<eigen_dense_t> eigensolver(eigen_dense_t(this->entries()));
232 
233  // eigenvalues and eigenvectors
234  eigen_vector_real_t evals = eigensolver.eigenvalues();
235  eigen_sparse_t evecs = eigensolver.eigenvectors().sparseView(1e-4,0.5);
236 
237  this->entries().setZero();
238  this->entries().reserve(evals.size());
239  for (eigen_idx_t idx = 0; idx < evals.size(); ++idx) {
240  this->entries().insert(idx, idx) = evals.coeffRef(idx);
241  }
242  this->entries().makeCompressed();
243 
244  this->basis() = (this->basis() * evecs).pruned(1e-4,0.5);
245  }
246 }

289 void Hamiltonianmatrix::doSerialization() {
290  if (bytes.size() == 0) {
291  entries_.makeCompressed();
292  basis_.makeCompressed();
293 
294  // convert matrix "entries" to vectors of primitive data types
295  byte_t entries_flags = 0;
296  if (entries_.IsRowMajor) {
297  entries_flags |= csr_not_csc;
298  }
299  if (utils::is_complex<scalar_t>::value) {
300  entries_flags |= complex_not_real;
301  }
302  storage_idx_t entries_rows = entries_.rows();
303  storage_idx_t entries_cols = entries_.cols();
304  std::vector<scalar_t> entries_data(entries_.valuePtr(), entries_.valuePtr()+entries_.nonZeros());
305  std::vector<storage_real_t> entries_data_real, entries_data_imag;
306  splitComplex(entries_data_real,entries_data_imag,entries_data);
307  std::vector<storage_idx_t> entries_indices(entries_.innerIndexPtr(), entries_.innerIndexPtr()+entries_.nonZeros());
308  std::vector<storage_idx_t> entries_indptr(entries_.outerIndexPtr(), entries_.outerIndexPtr()+entries_.outerSize());
309 
310  // convert matrix "basis" to vectors of primitive data types
311  byte_t basis_flags = 0;
312  if (basis_.IsRowMajor) {
313  basis_flags |= csr_not_csc;
314  }
315  if (utils::is_complex<scalar_t>::value) {
316  basis_flags |= complex_not_real;
317  }
318  storage_idx_t basis_rows = basis_.rows();
319  storage_idx_t basis_cols = basis_.cols();
320  std::vector<scalar_t> basis_data(basis_.valuePtr(), basis_.valuePtr()+basis_.nonZeros());
321  std::vector<storage_real_t> basis_data_real, basis_data_imag;
322  splitComplex(basis_data_real,basis_data_imag,basis_data);
323  std::vector<storage_idx_t> basis_indices(basis_.innerIndexPtr(), basis_.innerIndexPtr()+basis_.nonZeros());
324  std::vector<storage_idx_t> basis_indptr(basis_.outerIndexPtr(), basis_.outerIndexPtr()+basis_.outerSize());
325 
326  // serialize vectors of primitive data types
327  Serializer s;
328  s << entries_flags;
329  s << entries_rows;
330  s << entries_cols;
331  s << entries_data_real;
332  if (entries_flags & complex_not_real) s << entries_data_imag;
333  s << entries_indices;
334  s << entries_indptr;
335  s << basis_flags;
336  s << basis_rows;
337  s << basis_cols;
338  s << basis_data_real;
339  if (basis_flags & complex_not_real) s << basis_data_imag;
340  s << basis_indices;
341  s << basis_indptr;
342  s.save(bytes);
343  }

351 void Hamiltonianmatrix::doDeserialization() {
352  // deserialize vectors of primitive data types
353  byte_t entries_flags;
354  storage_idx_t entries_rows, entries_cols;
355  std::vector<storage_real_t> entries_data_real, entries_data_imag;
356  std::vector<idx_t> entries_indices;
357  std::vector<idx_t> entries_indptr;
358  byte_t basis_flags;
359  storage_idx_t basis_rows, basis_cols;
360  std::vector<storage_real_t> basis_data_real, basis_data_imag;
361  std::vector<idx_t> basis_indices;
362  std::vector<idx_t> basis_indptr;
363 
364  Serializer s;
365  s.load(bytes);
366  s >> entries_flags;
367  s >> entries_rows;
368  s >> entries_cols;
369  s >> entries_data_real;
370  if (entries_flags & complex_not_real) s >> entries_data_imag;
371  s >> entries_indices;
372  s >> entries_indptr;
373  s >> basis_flags;
374  s >> basis_rows;
375  s >> basis_cols;
376  s >> basis_data_real;
377  if (basis_flags & complex_not_real) s >> basis_data_imag;
378  s >> basis_indices;
379  s >> basis_indptr;
380 
381  if((((entries_flags & complex_not_real) > 0) != utils::is_complex<scalar_t>::value) ||
382  (((basis_flags & complex_not_real) > 0) != utils::is_complex<scalar_t>::value)) {
383  std::string msg("The data type used in the program does not fit the data type used in the serialized objects.");
384  std::cout << ">>ERR" << msg << std::endl;
385  throw std::runtime_error(msg);
386  }
387 
388  // build matrix "entries_"
389  std::vector<scalar_t> entries_data;
390  mergeComplex(entries_data_real,entries_data_imag,entries_data);
391  entries_ = eigen_sparse_t(entries_rows,entries_cols);
392  entries_.makeCompressed();
393  entries_.resizeNonZeros(entries_data.size());
394  std::copy(entries_data.begin(),entries_data.end(),entries_.valuePtr());
395  std::copy(entries_indices.begin(),entries_indices.end(),entries_.innerIndexPtr());
396  std::copy(entries_indptr.begin(),entries_indptr.end(),entries_.outerIndexPtr());
397  entries_.finalize();
398 
399  // build matrix "basis_"
400  std::vector<scalar_t> basis_data;
401  mergeComplex(basis_data_real,basis_data_imag,basis_data);
402  basis_ = eigen_sparse_t(basis_rows,basis_cols);
403  basis_.makeCompressed();
404  basis_.resizeNonZeros(basis_data.size());
405  std::copy(basis_data.begin(),basis_data.end(),basis_.valuePtr());
406  std::copy(basis_indices.begin(),basis_indices.end(),basis_.innerIndexPtr());
407  std::copy(basis_indptr.begin(),basis_indptr.end(),basis_.outerIndexPtr());
408  basis_.finalize();
409 }

467 Hamiltonianmatrix combine(const Hamiltonianmatrix &lhs, const Hamiltonianmatrix &rhs, const real_t &deltaE, std::shared_ptr<BasisnamesTwo> basis_two, const Symmetry &sym) {
468  // TODO program a faster method for samebasis == true
469 
470  size_t num_basisvectors = lhs.num_basisvectors()*rhs.num_basisvectors();
471  size_t num_coordinates = lhs.num_coordinates()*rhs.num_coordinates();
472 
473 
474  ////////////////////////////////////////////////////////
475  ////// Mapping used in case of reflection symmetry /////
476  ////////////////////////////////////////////////////////
477 
478  std::vector<size_t> mapping(num_coordinates, -1);
479  if (sym.reflection != NA) {
480  std::unordered_map<StateTwo, size_t> buffer;
481  for (auto state: *basis_two) {
482  if (state.m[0] < 0) {
483  continue;
484  }
485  state.m[0] *= -1;
486  state.m[1] *= -1;
487  buffer[state] = state.idx;
488  }
489  for (auto state: *basis_two) {
490  if (state.m[0] > 0) {
491  continue;
492  }
493  mapping[buffer[state]] = state.idx;
494  if (sym.inversion != NA || sym.permutation != NA ) mapping[state.idx] = buffer[state];
495  }
496  }
497 
498 
499  ////////////////////////////////////////////////////////
500  ////// Combine basis and entries ///////////////////////
501  ////////////////////////////////////////////////////////
502 
503  // This approach does only work if lhs.entries() and rhs.entries() are diagonal // TODO assert
504 
505  // === Initialize matrices ===
506  eigen_vector_t diag1 = lhs.entries().diagonal();
507  eigen_vector_t diag2 = rhs.entries().diagonal();
508 
509  // Number of elements for which space sould be reserved
510  size_t size_basis = num_basisvectors; // TODO estimate better
511  size_t size_entries = num_basisvectors;
512 
513  Hamiltonianmatrix mat(size_basis, size_entries);
514 
515  // === Combine basis and entries ===
516  size_t col = 0; // basis vector
517  for (eigen_idx_t col_1=0; col_1<lhs.basis().outerSize(); ++col_1) { // outerSize() == num_cols = num_basisvectors()
518  for (eigen_idx_t col_2=0; col_2<rhs.basis().outerSize(); ++col_2) {
519 
520  // In case of inversion symmetry: skip half of the basis vector pairs
521  if ((sym.inversion == EVEN && col_1 <= col_2) || // gerade
522  (sym.inversion == ODD && col_1 < col_2)) { // ungerade
523  continue;
524  }
525 
526  // In case of permutation symmetry: skip half of the basis vector pairs
527  if ((sym.permutation == EVEN && col_1 <= col_2) || // sym
528  (sym.permutation == ODD && col_1 < col_2)) { // asym
529  continue;
530  }
531 
532  // TODO combine inversion and permutation symmetry (one variable "permuinversion"), think about further simplifications
533 
534  // --- Combine diagonal elements for mat.entries() ---
535  scalar_t val_entries = diag1[col_1] + diag2[col_2]; // diag(V) x I + I x diag(V)
536 
537  // Check whether the new diagonal element is within the energy cutoff
538  if (std::abs(val_entries) < deltaE+1e-11 || deltaE < 0) { // TODO avoid the "+1e-11" hack
539 
540  // Variable that stores whether the combindes basis vector, that belongs to the combined diagonal element, is valid
541  bool existing = false;
542 
543  // --- Combine basis vectors for mat.basis() ---
544  for (eigen_iterator_t triple_1(lhs.basis(),col_1); triple_1; ++triple_1) {
545  for (eigen_iterator_t triple_2(rhs.basis(),col_2); triple_2; ++triple_2) {
546  size_t row = rhs.num_coordinates()*triple_1.row() + triple_2.row(); // coordinate
547 
548  // Get pair state that belongs to the current coordinate of the combined basis vector
549  const auto &state = basis_two->get(row);
550 
551  float M = state.m[0]+state.m[1];
552  int parityL = std::pow(-1, state.l[0] + state.l[1]);
553  int parityJ = std::pow(-1, state.j[0] + state.j[1]);
554  int parityM = std::pow(-1, state.m[0] + state.m[1]);
555 
556  // In case of inversion and reflection symmetry: check whether the inversion symmetric state is already reflection symmetric
557  bool skip_reflection = false;
558  if (sym.inversion != NA && col_1 != col_2 && sym.reflection != NA && mapping[row] == rhs.num_coordinates()*triple_2.row() + triple_1.row()) {
559  if ( ((sym.inversion == EVEN) ? -parityL : parityL) != ((sym.reflection == EVEN) ? parityL*parityJ*parityM : -parityL*parityJ*parityM) ) {
560  continue; // the parity under inversion and reflection is different
561  } else {
562  skip_reflection = true; // the parity under inversion and reflection is the same
563  }
564  }
565 
566  // In case of permutation and reflection symmetry: check whether the permutation symmetric state is already reflection symmetric
567  if (sym.permutation != NA && col_1 != col_2 && sym.reflection != NA && mapping[row] == rhs.num_coordinates()*triple_2.row() + triple_1.row()) {
568  if ( ((sym.permutation == EVEN) ? -1 : 1) != ((sym.reflection == EVEN) ? parityL*parityJ*parityM : -parityL*parityJ*parityM) ) {
569  continue; // the parity under permutation and reflection is different
570  } else {
571  skip_reflection = true; // the parity under permutation and reflection is the same
572  }
573  }
574 
575  // In case of inversion and permutation symmetry: the inversion symmetric state is already permutation symmetric
576  bool skip_permutation = false;
577  if (sym.inversion != NA && sym.permutation != NA && col_1 != col_2) {
578  if ( ((sym.inversion == EVEN) ? -parityL : parityL) != ((sym.permutation == EVEN) ? -1 : 1) ) {
579  continue; // the parity under inversion and permutation is different
580  } else {
581  skip_permutation = true; // the parity under inversion and permutation is the same
582  }
583  }
584 
585  // In case of rotation symmetry: skip coordinates with wrong total magnetic momentum
586  if(sym.rotation != NA && sym.rotation != M && (sym.reflection == NA || sym.rotation != -M)) {
587  continue;
588  }
589 
590  // In case of reflection symmetry: skip half of the coordinates
591  if (sym.reflection != NA && state.m[0] < 0 && !skip_reflection) {
592  continue;
593  }
594 
595  // Calculate coefficient that belongs to the current coordinate
596  scalar_t val_basis = triple_1.value() * triple_2.value(); // coefficient
597  if (sym.reflection != NA && !skip_reflection) {
598  val_basis /= std::sqrt(2);
599  }
600  if (sym.inversion != NA && col_1 != col_2) {
601  val_basis /= std::sqrt(2);
602  }
603  if (sym.permutation != NA && col_1 != col_2 && !skip_permutation) {
604  val_basis /= std::sqrt(2);
605  }
606 
607  // Save the coefficient taking into account the symmetrization
608  mat.addBasis(row,col,val_basis);
609 
610  if (sym.reflection != NA && !skip_reflection) {
611  size_t r = mapping[row];
612  scalar_t v = val_basis;
613  v *= (sym.reflection == EVEN) ? parityL*parityJ*parityM : -parityL*parityJ*parityM;
614  mat.addBasis(r,col,v);
615  }
616 
617  if (sym.inversion != NA && col_1 != col_2) {
618  size_t r = rhs.num_coordinates()*triple_2.row() + triple_1.row();
619  scalar_t v = val_basis;
620  v *= (sym.inversion == EVEN) ? -parityL : parityL;
621  mat.addBasis(r,col,v);
622  }
623 
624  if (sym.inversion != NA && col_1 != col_2 && sym.reflection != NA && !skip_reflection) {
625  size_t r = rhs.num_coordinates()*triple_2.row() + triple_1.row();
626  r = mapping[r];
627  scalar_t v = val_basis;
628  v *= (sym.reflection == EVEN) ? parityL*parityJ*parityM : -parityL*parityJ*parityM;
629  v *= (sym.inversion == EVEN) ? -parityL : parityL;
630  mat.addBasis(r,col,v);
631  }
632 
633  if (sym.permutation != NA && col_1 != col_2 && !skip_permutation) {
634  size_t r = rhs.num_coordinates()*triple_2.row() + triple_1.row();
635  scalar_t v = val_basis;
636  v *= (sym.permutation == EVEN) ? -1 : 1;
637  mat.addBasis(r,col,v);
638  }
639 
640  if (sym.permutation != NA && col_1 != col_2 && !skip_permutation && sym.reflection != NA && !skip_reflection) {
641  size_t r = rhs.num_coordinates()*triple_2.row() + triple_1.row();
642  r = mapping[r];
643  scalar_t v = val_basis;
644  v *= (sym.reflection == EVEN) ? parityL*parityJ*parityM : -parityL*parityJ*parityM;
645  v *= (sym.permutation == EVEN) ? -1 : 1;
646  mat.addBasis(r,col,v);
647  }
648 
649  existing = true;
650  }
651  }
652 
653  // Save the combined diagonal element if the corresponding combined basis vector is valid
654  if (existing) {
655  mat.addEntries(col,col,val_entries);
656  ++col;
657  }
658  }
659  }
660  }
661 
662  // Finalize the Hamiltonian matrix
663  num_basisvectors = col;
664  mat.compress(num_basisvectors, num_coordinates);
665  return mat;
666 }
