30 /** \brief Quantum defect storage
31  *
32  * The quantum defect queries a database for Rydberg-Ritz coefficients and
33  * model potential parameters. The Rydberg-Ritz coefficients are not exposed to
34  * the outside but included in the energy data member.
35  *
36  * The slight deviation from the hydrogen atom for Rydberg atoms are captured
37  * in the so-called quantum defect. The hydrogen energy is given by \f$ E =
38  * \mathrm{Ry}/n^2 \f$. For Rydberg atoms this is slightly modified to \f$ E^*
39  * = \mathrm{Ry}/n^{*2} \f$ where \f$ n^* = n - \delta_{nlj} \f$. This
40  * correction can be expanded in a series
41  * \f[
42  * \delta_{nlj} = \delta_0
43  * + \frac{\delta_2}{(n-\delta_0)^2}
44  * + \frac{\delta_4}{(n-\delta_0)^4}
45  * + \frac{\delta_6}{(n-\delta_0)^6}
46  * + \cdots \;
47  * \f]
48  * The coefficients of the series \f$ \delta_i \f$ (Rydberg-Ritz coefficients)
49  * are loaded from the database.
50  *
51  * The object also loads the parameters for the model potentials which are used
52  * in the Numerov object.
53  */

64  /** \brief Comparison for keys */
65  bool operator==(Key const &o) const
66  {
67  return (species == o.species && n == o.n && l == o.l && j == o.j);
68  }
69  };

71  //** \brief Hash for Key */
72  struct Hash {
73  /** \brief Hash the key using boost::hash_combine */
74  std::size_t operator()(Key const &key) const
75  {
76  std::size_t seed = 0;
77  boost::hash_combine(seed, key.species);
78  boost::hash_combine(seed, key.n);
79  boost::hash_combine(seed, key.l);
80  boost::hash_combine(seed, key.j);
81  return seed;
82  }
83  };
