49 /** \brief Generate a random integer
50  *
51  * This is very similar to the implementation of randint in the GCC
52  * standard library Fundamentals TS v2. It is specified by the
53  * standard per clause 13.2.2.1, Function template randint.
54  *
55  * The function generates a random integer in the closed interval
56  * [\p a,\p b].
57  *
58  * \param a lower bound
59  * \param b upper bound
60  * \returns random integer between \p a and \p b.
61  */
62 template <typename T>
63 inline T randint(T a, T b)
64 {
65  static_assert(std::is_integral<T>::value && sizeof(T) > 1,
66  "The type must be an integer!");
67  return std::uniform_int_distribution<T>(a, b)(randint_engine());
68 }

71 inline uint64_t FNV64(uint8_t *s, size_t sz)
72 {
73  const uint64_t magicPrime = 0x00000100000001b3;
74  uint64_t hash = 0xcbf29ce484222325;
75 
76  for (size_t i = 0; i < sz; ++i) {
77  hash = (hash ^ s[i]) * magicPrime;
78  }
