65 void QuantumDefect::setup(sqlite3 *db)
66 {
67  static Cache<Key, Element, Hash> cache;
68 
69  Key const key{species, n, l, j};
70  if (auto oe = cache.restore(key))
71  {
72  e = oe.get(); // Restore cache
73  return; // Return early
74  }
75 
76  std::stringstream ss;
77  sqlite::statement stmt(db);
78  int pot_max_l, ryd_max_l;
79  int pot_l, ryd_l;
80  double ryd_max_j;
81  double ryd_j;
82 
83  // Determine maximal L for model potentials
84  stmt.set("select MAX(L) from model_potential where (element = ?);");
85  stmt.prepare();
86  stmt.bind(1, species);
87  if (stmt.step())
88  pot_max_l = stmt.get<int>(0);
89  else
90  throw no_defect(*this);
91 
92  // The l to be used is the minimum of the two below
93  pot_l = std::min(l, pot_max_l);
94 
95  // Determine maximal L for Rydberg-Ritz coefficients
96  stmt.reset();
97  stmt.set("select MAX(L) from rydberg_ritz where (element = ?);");
98  stmt.prepare();
99  stmt.bind(1, species);
100  if (stmt.step())
101  ryd_max_l = stmt.get<int>(0);
102  else
103  throw no_defect(*this);
104 
105  // The l to be used is the minimum of the two below
106  ryd_l = std::min(l, ryd_max_l);
107 
108  // Determine maximal J for Rydberg-Ritz coefficients
109  stmt.reset();
110  stmt.set("select MAX(J) from rydberg_ritz where ((element = ?1) and (L = "
111  "?2));");
112  stmt.prepare();
113  stmt.bind(1, species);
114  stmt.bind(2, ryd_l);
115  if (stmt.step())
116  ryd_max_j = stmt.get<double>(0);
117  else
118  throw no_defect(*this);
119 
120  // The j to be used is the minimum of the two below
121  ryd_j = std::min(j, ryd_max_j);
122 
123  // Load model potentials from database
124  stmt.reset();
125  stmt.set("select ac,Z,a1,a2,a3,a4,rc from model_potential where ((element "
126  "= ?1) and (L = ?2));");
127  stmt.prepare();
128  stmt.bind(1, species);
129  stmt.bind(2, pot_l);
130  if (stmt.step()) {
131  e.ac = stmt.get<double>(0);
132  e.Z = stmt.get<int>(1);
133  e.a1 = stmt.get<double>(2);
134  e.a2 = stmt.get<double>(3);
135  e.a3 = stmt.get<double>(4);
136  e.a4 = stmt.get<double>(5);
137  e.rc = stmt.get<double>(6);
138  } else
139  throw no_defect(*this);
140 
141  // Load Rydberg-Ritz coefficients from database
142  stmt.reset();
143  stmt.set("select d0,d2,d4,d6,d8,Ry from rydberg_ritz where ((element = ?1) "
144  "and (L = ?2) and (J = ?3));");
145  stmt.prepare();
146  stmt.bind(1, species);
147  stmt.bind(2, ryd_l);
148  stmt.bind(3, ryd_j);
149  e.nstar = n;
150  double Ry_inf = 109737.31568525;
151  double Ry = Ry_inf;
152  if (stmt.step()) {
153  double d0 = stmt.get<double>(0);
154  double d2 = stmt.get<double>(1);
155  double d4 = stmt.get<double>(2);
156  double d6 = stmt.get<double>(3);
157  double d8 = stmt.get<double>(4);
158  Ry = stmt.get<double>(5);
159  e.nstar -= d0 + d2 / pow(n - d0, 2) + d4 / pow(n - d0, 4) +
160  d6 / pow(n - d0, 6) + d8 / pow(n - d0, 8);
161  } else
162  throw no_defect(*this);
163 
164  e.energy = -.5 * (Ry / Ry_inf) / (e.nstar * e.nstar) * au2GHz;
165 
166  cache.save(key, e);
167 }
