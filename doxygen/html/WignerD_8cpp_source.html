35 double WignerD::evalWignerdPiHalf(float j, float m, float mp) {
36  double r = 0;
37  for (unsigned int k = std::max(0, static_cast<int>(mp - m)); k <= j + std::min(mp, - m); ++k) {
38  r += std::pow(-1, k) * boost::math::binomial_coefficient<double>(j + mp, k) * boost::math::binomial_coefficient<double>(j - mp, k + m - mp);
39  }
40  r *= std::pow(-1., m-mp) / std::pow(2., j) * std::sqrt(boost::math::factorial<double>(j + m) * boost::math::factorial<double>(j - m) / (boost::math::factorial<double>(j + mp) * boost::math::factorial<double>(j - mp)));
41  return r;
42 }

44 double WignerD::evalWignerd(float j, float m, float mp, double beta) {
45  std::complex<double> r = 0;
46  for (float mpp = j; mpp >= -j; --mpp) { // TODO parallelize
47  r += this->evalWignerdPiHalf(j, m, mpp) * std::complex<double>(std::cos(-mpp * beta), std::sin(-mpp * beta)) * this->evalWignerdPiHalf(j, mpp, -mp);
48  }
49  r *= std::pow(std::complex<double>(0, 1), 2. * j - m - mp) * std::pow(-1., 2. * m);
50  return r.real();
