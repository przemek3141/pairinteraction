22 template<typename T>
23 bool is_sorted(const T& mat) {
24  for(Index k = 0; k<mat.outerSize(); ++k)
25  {
26  Index prev = -1;
27  for(typename T::InnerIterator it(mat,k); it; ++it)
28  {
29  if(prev>=it.index())
30  return false;
31  prev = it.index();
32  }
33  }
34  return true;
35 }
