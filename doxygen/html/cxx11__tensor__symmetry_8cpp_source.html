48 // helper class for checking that the symmetry groups are correct
49 struct checkIdx {
50  template<typename ArrType>
51  static inline int doCheck_(ArrType e, int flags, int dummy, std::set<uint64_t>& found, std::map<uint64_t, int> const& expected)
52  {
53  // use decimal representation of value
54  uint64_t value = e[0];
55  for (std::size_t i = 1; i < e.size(); i++)
56  value = value * 10 + e[i];
57 
58  // we want to make sure that we find each element
59  auto it = expected.find(value);
60  VERIFY((it != expected.end()));
61  VERIFY_IS_EQUAL(it->second, flags);
62 
63  // we want to make sure we only have each element once;
64  // set::insert returns true for the second part of the pair
65  // if the element was really inserted and not already there
66  auto p = found.insert(value);
67  VERIFY((p.second));
68 
69  return dummy;
70  }
71 
72  static inline int run(std::vector<int> e, int flags, int dummy, std::set<uint64_t>& found, std::map<uint64_t, int> const& expected)
73  {
74  return doCheck_(e, flags, dummy, found, expected);
75  }

77  template<std::size_t N>
78  static inline int run(std::array<int, N> e, int flags, int dummy, std::set<uint64_t>& found, std::map<uint64_t, int> const& expected)
79  {
80  return doCheck_(e, flags, dummy, found, expected);
81  }
