87  template<class T>
88  void addRotated(const StateTwo &state, const size_t &idx, std::vector<Eigen::Triplet<T>> &triplets, WignerD &wigner, const double &alpha, const double &beta, const double &gamma) {
89  // Check whether the angles are compatible to the used data type
90  double tolerance = 1e-16;
91  if (std::is_same<T, double>::value && std::abs(std::remainder(alpha,2*M_PI)) > tolerance) throw std::runtime_error( "If the Euler angle alpha is not a multiple of 2 pi, the Wigner D matrix element is complex and cannot be converted to double." );
92  if (std::is_same<T, double>::value && std::abs(std::remainder(gamma,2*M_PI)) > tolerance) throw std::runtime_error( "If the Euler angle gamma is not a multiple of 2 pi, the Wigner D matrix element is complex and cannot be converted to double." );
93 
94  // Add rotated triplet entries
95  StateTwo newstate = state;
96  std::vector<T> val2_vector;
97  val2_vector.reserve(2*state.second().j+1);
98 
99  for (float m2 = -state.second().j; m2 <= state.second().j; ++m2) {
100  val2_vector.push_back(convert<T>(wigner(state.second().j, state.second().m, m2, alpha, beta, gamma)));
101  }
102 
103  for (float m1 = -state.first().j; m1 <= state.first().j; ++m1) {
104  T val1 = convert<T>(wigner(state.first().j, state.first().m, m1, alpha, beta, gamma));
105 
106  for (float m2 = -state.second().j; m2 <= state.second().j; ++m2) {
107  newstate.m = {{m1, m2}};
108  auto state_iter = states.get<1>().find(newstate);
109 
110  if (state_iter != states.get<1>().end()) {
111  T val = val1*val2_vector[m2+state.second().j];
112  triplets.push_back(Eigen::Triplet<T>(state_iter->idx, idx, val));
113  } else {
114  std::cerr << "Warning: Incomplete rotation because the basis is lacking some Zeeman levels." << std::endl;
115  }
116  }
117  }
118  }
