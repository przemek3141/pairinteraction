300  kernel.packet[2].x = tmp;

304  kernel.packet[3].x = tmp;

306  tmp = kernel.packet[1].z;

308  kernel.packet[2].y = tmp;

310  tmp = kernel.packet[1].w;

312  kernel.packet[3].y = tmp;
