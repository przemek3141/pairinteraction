26 /** \brief Generic cache object
27  *
28  * This generic cache object strives to provide a thread-safe cache
29  * based on a `std::unordered_map` protected from smashing by a mutex.
30  *
31  * There are only few public methods, which are save, restore, and
32  * clear. Iteration over the cache was left out intentionally as the
33  * user should not search the cache manually for entries.
34  */
35 template <typename Key, typename Element, typename Hash = std::hash<Key>>
36 class Cache
37 {
38  typedef std::unordered_map<Key, Element, Hash> cache_t;
39 
40  cache_t cache;
41  std::mutex cache_mutex;
42 
43 public:
44  /** \brief Save something in the cache
45  *
46  * To prevent race-conditions this function will throw if the
47  * element is already in the cache.
48  *
49  * The usage is extremely straight-forward
50  * \code
51  * cache.save(key,element);
52  * \endcode
53  *
54  * \param key Key
55  * \param e Element
56  * \throws std::runtime_error if the element is already in the cache
57  */
58  void save(Key const &key, Element const &e)
59  {
60  std::lock_guard<std::mutex> lock(cache_mutex);
61  if (!cache.emplace(std::make_pair(key, e)).second) {
62  throw std::runtime_error("Cache smashing detected!");
63  }
64  }
65 
66  /** \brief Restore something from the cache
67  *
68  * When restoring from the cache, the situation might occur that
69  * an element is not yet available in the cache. We do not want
70  * to throw an exception in that case because that case might be
71  * very frequent when building the cache. Therefore restoring
72  * returns an optional value, i.e. it might not void. Hence the
73  * access pattern is also a little different.
74  *
75  * \code
76  * if (auto optional_element = cache.restore(key,element)) {
77  * // Cache hit! Restore cache...
78  * element = optional_element.get();
79  * } else {
80  * // Cache miss :-(
81  * }
82  * \endcode
83  *
84  * \param key Key
85  * \returns Optional element
86  */
87  boost::optional<Element> restore(Key const &key)
88  {
89  std::lock_guard<std::mutex> lock(cache_mutex);
90  auto cached_it = cache.find(key);
91  if (cached_it != cache.end()) {
92  return cached_it->second;
93  }
94  return boost::none;
95  }
96 
97  /** \brief Clear the cache
98  *
99  * Delete all elements in the cache
100  */
101  void clear() { cache.clear(); }
102 };
