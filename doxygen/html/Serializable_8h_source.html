61  void save(bytes_t &bytes) {
62  size_t nItems = buffer_types.size();
63 
64  // clear and resize bytes
65  bytes.clear();
66  bytes.resize(size());
67 
68  // save items to bytes
69  pbytes = bytes.begin();
70 
71  for (size_t i = 0; i < nItems; ++i) {
72  if (buffer_isVector[i]) {
73  serialize(&type_ids[std::type_index(typeid(storage_idx_t))], sizeof(type_t));
74  serialize(&buffer_nums[i], sizeof(storage_idx_t));
75  }
76 
77  serialize(&buffer_types[i], sizeof(type_t));
78  serialize(buffer_pitems[i], buffer_nums[i]*buffer_sizes[i]);
79  }
80 
81  // the buffers are not needed anymore
82  buffer_isVector.clear();
83  buffer_types.clear();
84  buffer_pitems.clear();
85  buffer_nums.clear();
86  buffer_sizes.clear();
87  }

89  size_t position() {
90  size_t sz = size();
91  if (sz > 0) {
92  return sz;
93  } else {
94  return std::distance(cpbytes_start,cpbytes);
95  }
96  }

129  size_t size() {
130  size_t nItems = buffer_types.size();
131  size_t totalsize = 0;
132 
133  for (size_t i = 0; i < nItems; ++i) {
134  if (buffer_isVector[i]) {
135  totalsize += sizeof(type_t);
136  totalsize += sizeof(storage_idx_t);
137  }
138  totalsize += sizeof(type_t);
139  totalsize += buffer_nums[i]*buffer_sizes[i];
140  }
141 
142  return totalsize;
143  }
