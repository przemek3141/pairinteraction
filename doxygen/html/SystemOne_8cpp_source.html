134 void SystemOne::initializeBasis()
135 {
136  // If the basis is infinite, throw an error
137  if (range_n.empty() && (energy_min == std::numeric_limits<double>::lowest() || energy_max == std::numeric_limits<double>::max())) {
138  throw std::runtime_error( "The number of basis elements is infinite. The basis has to be restricted." );
139  }
140 
141  ////////////////////////////////////////////////////////////////////
142  /// Build one atom states //////////////////////////////////////////
143  ////////////////////////////////////////////////////////////////////
144 
145  // TODO check whether symmetries are applicable
146 
147  size_t idx = 0;
148  std::vector<eigen_triplet_t> coefficients_triplets; // TODO reserve states, coefficients_triplets, hamiltonianmatrix_triplets
149  std::vector<eigen_triplet_t> hamiltonianmatrix_triplets;
150  std::set<int> range_adapted_n, range_adapted_l;
151  std::set<float> range_adapted_j, range_adapted_m;
152 
153  if (range_n.empty()) {
154  throw std::runtime_error( "The calculation of range_n via energy restrictions is not yet implemented." ); // TODO
155  } else {
156  range_adapted_n = range_n;
157  }
158  for (auto n : range_adapted_n) {
159 
160  if (range_l.empty()) {
161  this->range(range_adapted_l, 0, n-1);
162  } else {
163  range_adapted_l = range_l;
164  }
165  for (auto l : range_adapted_l) {
166  if (l > n-1) continue;
167 
168  if (range_j.empty()) {
169  range_adapted_j = {std::fabs(l-0.5f), l+0.5f};
170  } else {
171  range_adapted_j = range_j;
172  }
173  for (auto j : range_adapted_j) {
174  if (std::fabs(j-l) != 0.5) continue;
175 
176  double energy = StateOne(element,n,l,j,0.5).getEnergy();
177  if (!checkIsEnergyValid(energy)) continue;
178 
179  if (range_m.empty()) {
180  this->range(range_adapted_m, -j, j);
181  } else {
182  range_adapted_m = range_m;
183  }
184 
185  // Consider rotation symmetry
186  std::set<float> range_allowed_m;
187  if (sym_rotation.count(static_cast<float>(ARB)) == 0) {
188  std::set_intersection(sym_rotation.begin(), sym_rotation.end(), range_adapted_m.begin(), range_adapted_m.end(), std::inserter(range_allowed_m, range_allowed_m.begin()));
189  } else {
190  range_allowed_m = range_adapted_m;
191  }
192 
193  for (auto m : range_allowed_m) {
194  if (std::fabs(m) > j) continue;
195 
196  // In case of reflection symmetry, skip half of the basis vectors
197  if (sym_reflection != NA && m < 0) continue;
198 
199  // Store the energy of the unperturbed one atom state
200  hamiltonianmatrix_triplets.push_back(eigen_triplet_t(idx,idx,energy));
201 
202  // Adapt the normalization if required by symmetries
203  scalar_t value = 1;
204  if (sym_reflection != NA) {
205  value /= std::sqrt(2);
206  }
207 
208  // Add an entry to the current basis vector
209  this->addCoefficient(StateOne(element,n,l,j,m), idx, value, coefficients_triplets);
210 
211  // Add further entries to the current basis vector if required by symmetries
212  if (sym_reflection != NA) {
213  value *= (sym_reflection == EVEN) ? std::pow(-1,l+m-j)*this->imaginaryUnit<scalar_t>() : -std::pow(-1,l+m-j)*this->imaginaryUnit<scalar_t>(); // S_y is invariant under reflection through xz-plane
214  this->addCoefficient(StateOne(element,n,l,j,-m), idx, value, coefficients_triplets);
215  }
216 
217  ++idx;
218  }
219  }
220  }
221  }
222 
223  // Build data
224  coefficients.resize(states.size(),idx);
225  coefficients.setFromTriplets(coefficients_triplets.begin(), coefficients_triplets.end());
226  coefficients_triplets.clear();
227 
228  hamiltonianmatrix.resize(idx,idx);
229  hamiltonianmatrix.setFromTriplets(hamiltonianmatrix_triplets.begin(), hamiltonianmatrix_triplets.end());
230  hamiltonianmatrix_triplets.clear();
231 }

237 void SystemOne::initializeInteraction() {
238  ////////////////////////////////////////////////////////////////////
239  /// Prepare the calculation of the interaction /////////////////////
240  ////////////////////////////////////////////////////////////////////
241 
242  // Check if something to do
243  double tolerance = 1e-24;
244 
245  std::vector<int> erange, brange;
246  std::vector<std::array<int, 2>> drange;
247 
248  for (const auto &entry : efield_spherical) {
249  if (std::abs(entry.second) > tolerance && interaction_efield.find(-entry.first) == interaction_efield.end()) {
250  erange.push_back(-entry.first); // TODO use entry.first without a minus
251  }
252  }
253  for (const auto &entry : bfield_spherical) {
254  if (std::abs(entry.second) > tolerance && interaction_bfield.find(-entry.first) == interaction_bfield.end()) {
255  brange.push_back(-entry.first); // TODO use entry.first without a minus
256  }
257  }
258  for (const auto &entry : diamagnetism_terms) {
259  if (diamagnetism && std::abs(entry.second) > tolerance && interaction_diamagnetism.find(entry.first) == interaction_diamagnetism.end()) {
260  drange.push_back(entry.first);
261  }
262  }
263 
264  if (erange.empty() && brange.empty() && drange.empty()) return;
265 
266  // Precalculate matrix elements
267  std::string matrixelementsdir = "";
268  if (!cachedir.empty()) matrixelementsdir = (cachedir / "cache_elements.db").string(); // TODO do this in the MatrixElements class, just pass cachedir as an argument to the constructor
269 
270  std::string tmp(element.begin(), element.end()); // TODO think of a better solution
271  MatrixElements matrixelements(tmp, matrixelementsdir);
272  auto states_converted = this->getStates(); // TODO remove
273  for (const auto &i : erange) matrixelements.precalculateElectricMomentum(states_converted, i);
274  for (const auto &i : brange) matrixelements.precalculateMagneticMomentum(states_converted, i);
275  for (const auto &i : drange) matrixelements.precalculateDiamagnetism(states_converted, i[0], i[1]);
276 
277  ////////////////////////////////////////////////////////////////////
278  /// Calculate the interaction in the canonical basis ///////////////
279  ////////////////////////////////////////////////////////////////////
280 
281  std::unordered_map<int, std::vector<eigen_triplet_t>> interaction_efield_triplets; // TODO reserve
282  std::unordered_map<int, std::vector<eigen_triplet_t>> interaction_bfield_triplets;// TODO reserve
283  std::unordered_map<std::array<int, 2>, std::vector<eigen_triplet_t>> interaction_diamagnetism_triplets; // TODO reserve
284 
285  // Loop over column entries
286  for (const auto &c: states) { // TODO parallelization
287 
288  if (c.state.element.empty()) continue; // TODO artifical states TODO [dummystates]
289 
290  // Loop over row entries
291  for (const auto &r: states) {
292 
293  if (r.state.element.empty()) continue; // TODO artifical states TODO [dummystates]
294  //if (r.idx < c.idx) continue; // TODO modify addTriplet so that skipping half of the entries work
295 
296  // E-field interaction
297  for (const auto &i : erange) {
298  if (selectionRulesMultipole(r.state, c.state, 1, i)) {
299  scalar_t value = matrixelements.getElectricMomentum(r.state, c.state);
300  this->addTriplet(interaction_efield_triplets[i], r.idx, c.idx, value);
301  break; // because for the other operators, the selection rule for the magnetic quantum numbers will not be fulfilled
302  }
303  }
304 
305  // B-field interaction
306  for (const auto &i : brange) {
307  if (selectionRulesMomentum(r.state, c.state, i)) {
308  scalar_t value = matrixelements.getMagneticMomentum(r.state, c.state);
309  this->addTriplet(interaction_bfield_triplets[i], r.idx, c.idx, value);
310  break; // because for the other operators, the selection rule for the magnetic quantum numbers will not be fulfilled
311  }
312  }
313 
314  // Diamagnetic interaction
315  for (const auto &i : drange) {
316  if (selectionRulesMultipole(r.state, c.state, i[0], i[1])) {
317  scalar_t value = matrixelements.getDiamagnetism(r.state, c.state, i[0]);
318  this->addTriplet(interaction_diamagnetism_triplets[i], r.idx, c.idx, value);
319  }
320  }
321  }
322  }
323 
324  ////////////////////////////////////////////////////////////////////
325  /// Build and transform the interaction to the used basis //////////
326  ////////////////////////////////////////////////////////////////////
327 
328  for (const auto &i : erange) {
329  interaction_efield[i].resize(states.size(),states.size());
330  interaction_efield[i].setFromTriplets(interaction_efield_triplets[i].begin(), interaction_efield_triplets[i].end());
331  interaction_efield_triplets[i].clear();
332 
333  interaction_efield[i] = coefficients.adjoint()*interaction_efield[i]*coefficients;
334  }
335 
336  for (const auto &i : brange) {
337  interaction_bfield[i].resize(states.size(),states.size());
338  interaction_bfield[i].setFromTriplets(interaction_bfield_triplets[i].begin(), interaction_bfield_triplets[i].end());
339  interaction_bfield_triplets[i].clear();
340 
341  interaction_bfield[i] = coefficients.adjoint()*interaction_bfield[i]*coefficients;
342  }
343 
344  for (const auto &i : drange) {
345  interaction_diamagnetism[i].resize(states.size(),states.size());
346  interaction_diamagnetism[i].setFromTriplets(interaction_diamagnetism_triplets[i].begin(), interaction_diamagnetism_triplets[i].end());
347  interaction_diamagnetism_triplets[i].clear();
348 
349  interaction_diamagnetism[i] = coefficients.adjoint()*interaction_diamagnetism[i]*coefficients;
350  }
351 }

357 void SystemOne::addInteraction() {
358  // Build the total Hamiltonian
359  double tolerance = 1e-24;
360 
361  if (std::abs(efield_spherical[+0]) > tolerance) hamiltonianmatrix -= interaction_efield[+0]*efield_spherical[+0];
362  if (std::abs(efield_spherical[-1]) > tolerance) hamiltonianmatrix += interaction_efield[+1]*efield_spherical[-1];
363  if (std::abs(efield_spherical[+1]) > tolerance) hamiltonianmatrix += interaction_efield[-1]*efield_spherical[+1];
364  if (std::abs(bfield_spherical[+0]) > tolerance) hamiltonianmatrix += interaction_bfield[+0]*bfield_spherical[+0];
365  if (std::abs(bfield_spherical[-1]) > tolerance) hamiltonianmatrix -= interaction_bfield[+1]*bfield_spherical[-1];
366  if (std::abs(bfield_spherical[+1]) > tolerance) hamiltonianmatrix -= interaction_bfield[-1]*bfield_spherical[+1];
367 
368  if (diamagnetism && std::abs(diamagnetism_terms[{{0,+0}}]) > tolerance) hamiltonianmatrix += interaction_diamagnetism[{{0,+0}}]*diamagnetism_terms[{{0,+0}}];
369  if (diamagnetism && std::abs(diamagnetism_terms[{{2,+0}}]) > tolerance) hamiltonianmatrix -= interaction_diamagnetism[{{2,+0}}]*diamagnetism_terms[{{2,+0}}];
370  if (diamagnetism && std::abs(diamagnetism_terms[{{2,+1}}]) > tolerance) hamiltonianmatrix += interaction_diamagnetism[{{2,+1}}]*diamagnetism_terms[{{2,+1}}]*std::sqrt(3);
371  if (diamagnetism && std::abs(diamagnetism_terms[{{2,-1}}]) > tolerance) hamiltonianmatrix += interaction_diamagnetism[{{2,-1}}]*diamagnetism_terms[{{2,-1}}]*std::sqrt(3);
372  if (diamagnetism && std::abs(diamagnetism_terms[{{2,+2}}]) > tolerance) hamiltonianmatrix -= interaction_diamagnetism[{{2,+2}}]*diamagnetism_terms[{{2,+2}}]*std::sqrt(1.5);
373  if (diamagnetism && std::abs(diamagnetism_terms[{{2,-2}}]) > tolerance) hamiltonianmatrix -= interaction_diamagnetism[{{2,-2}}]*diamagnetism_terms[{{2,-2}}]*std::sqrt(1.5);
374 }

444 void SystemOne::incorporate(SystemBase<StateOne> &system) {
445  // Combine parameters
446  if (element != dynamic_cast<SystemOne&>(system).element) throw std::runtime_error("The value of the variable 'element' must be the same for both systems.");
447  if (efield != dynamic_cast<SystemOne&>(system).efield) throw std::runtime_error("The value of the variable 'distance' must be the same for both systems."); // implies that efield_spherical is the same, too
448  if (bfield != dynamic_cast<SystemOne&>(system).bfield) throw std::runtime_error("The value of the variable 'angle' must be the same for both systems."); // implies that bfield_spherical is the same, too
449  if (diamagnetism != dynamic_cast<SystemOne&>(system).diamagnetism) throw std::runtime_error("The value of the variable 'ordermax' must be the same for both systems.");
450 
451  // Combine symmetries
452  unsigned int num_different_symmetries = 0;
453  if (sym_reflection != dynamic_cast<SystemOne&>(system).sym_reflection) {
454  sym_reflection = NA;
455  ++num_different_symmetries;
456  }
457  if (!std::equal(sym_rotation.begin(), sym_rotation.end(), dynamic_cast<SystemOne&>(system).sym_rotation.begin())) {
458  if (sym_rotation.count(static_cast<float>(ARB))!=0 || dynamic_cast<SystemOne&>(system).sym_rotation.count(static_cast<float>(ARB))!=0) {
459  sym_rotation = {static_cast<float>(ARB)};
460  } else {
461  sym_rotation.insert(dynamic_cast<SystemOne&>(system).sym_rotation.begin(), dynamic_cast<SystemOne&>(system).sym_rotation.end());
462  }
463  ++num_different_symmetries;
464  }
465  if (num_different_symmetries > 1) std::cerr << "Warning: The systems differ in more than one symmetry. For the combined system, the notion of symmetries might be meaningless." << std::endl;
466 
467  // Clear cached interaction
468  this->deleteInteraction();
469 }
