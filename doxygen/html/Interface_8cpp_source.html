59 int compute(const std::string &config_name, const std::string &output_name, std::string const &endpoint) {
60  zmq::endpoint::name = endpoint;
61 
62  auto context = zmq::context();
63  auto publisher = context.socket(ZMQ_PUB);
64  publisher.connect(zmq::endpoint::name.c_str());
65 
66  std::cout << std::unitbuf;
67 
68  Eigen::setNbThreads(1); // TODO set it to setNbThreads(0) when Eigen's multithreading is needed
69 
70  boost::filesystem::path path_config = boost::filesystem::absolute(config_name);
71  boost::filesystem::path path_cache = boost::filesystem::absolute(output_name);
72 
73  // === Load configuration ===
74  Configuration config;
75  config.load_from_json(path_config.string());
76 
77  bool existAtom1 = config.count("species1") && config.count("n1") && config.count("l1") && config.count("j1") && config.count("m1");
78  bool existAtom2 = config.count("species2") && config.count("n2") && config.count("l2") && config.count("j2") && config.count("m2");
79 
80  // === Solve the system ===
81  bool combined = config["samebasis"].str() == "true";
82 
83  if (combined) {
84  if (config["species1"].str() != config["species2"].str()) {
85  std::cout << "species1 and species2 has to be the same in order to use the same basis set." << std::endl;
86  return 1;
87  }
88  std::shared_ptr<HamiltonianOne> hamiltonian_one;
89  if (existAtom1 && existAtom2) {
90  publisher.send(">>TYP%7d", 3);
91  auto basisnames_one = std::make_shared<BasisnamesOne>(BasisnamesOne::fromBoth(config));
92  hamiltonian_one = std::make_shared<HamiltonianOne>(config, path_cache, basisnames_one);
93  }
94  std::shared_ptr<HamiltonianTwo> hamiltonian_two;
95  if (existAtom1 && existAtom2 && config.count("minR")) {
96  publisher.send(">>TYP%7d", 2);
97  hamiltonian_two = std::make_shared<HamiltonianTwo>(config, path_cache, hamiltonian_one);
98  }
99  } else {
100  std::shared_ptr<HamiltonianOne> hamiltonian_one1;
101  if (existAtom1) {
102  publisher.send(">>TYP%7d", 0);
103  auto basisnames_one1 = std::make_shared<BasisnamesOne>(BasisnamesOne::fromFirst(config));
104  hamiltonian_one1 = std::make_shared<HamiltonianOne>(config, path_cache, basisnames_one1);
105  }
106  std::shared_ptr<HamiltonianOne> hamiltonian_one2;
107  if (existAtom2) {
108  publisher.send(">>TYP%7d", 1);
109  auto basisnames_one2 = std::make_shared<BasisnamesOne>(BasisnamesOne::fromSecond(config));
110  hamiltonian_one2 = std::make_shared<HamiltonianOne>(config, path_cache, basisnames_one2);
111  }
112  std::shared_ptr<HamiltonianTwo> hamiltonian_two;
113  if (existAtom1 && existAtom2 && config.count("minR")) {
114  publisher.send(">>TYP%7d", 2);
115  hamiltonian_two = std::make_shared<HamiltonianTwo>(config, path_cache, hamiltonian_one1, hamiltonian_one2);
116  }
117  }
118 
119  // === Communicate that everything has finished ===
120  publisher.send(">>END");
121 
122  return 0;
123 }
