28  /** \brief Calculate the model potentials.
29  *
30  * This function calculates the model potentials and returns \f$
31  * V_{\mathrm{mod}}(r) = V_{\mathrm{C}}(r) + V_{\mathrm{P}}(r) +
32  * V_{\mathrm{s.o.}}(r) \f$
33  *
34  * \f[
35  * \begin{aligned}
36  * V_{\mathrm{C}}(r) &= - \frac{e^2}{4\pi\varepsilon_0} \frac{1 + (Z-1)\mathrm{e}^{-\alpha_1 r} - r(\alpha_3+\alpha_4 r)\mathrm{e}^{-\alpha_2 r}}{r} \\
37  * V_{\mathrm{P}}(r) &= -\frac{e^2}{(4\pi\varepsilon_0)^2} \frac{\alpha_d}{2 r^4} \left[ 1 - \mathrm{e}^{-(r/r_c)^6} \right] \\
38  * V_{\mathrm{s.o.}}(r) &= \frac{1}{2} \left( \frac{e^2}{4\pi\varepsilon_0} \right) \left( \frac{g_s}{2 m_e^2 c^2} \right) \frac{\boldsymbol{l}\cdot\boldsymbol{s}}{r^3}
39  * \end{aligned}
40  * \f]
41  *
42  * \param[in] qd Quantum defect data (parameters for the potentials)
43  * \param[in] x Radial position
44  * \returns Model potentials evaluted at position \p x with parameters \p qd
45  */

64 /** \brief %Numerov's method
65  *
66  * This class implements %Numerov's method using a couple of helper functions.
67  * %Numerov's method solves a differential equation of the form
68  * \f[
69  * y''(x) + g(x)y(x) = 0
70  * \f]
71  * The equation is solved by iteratively computing
72  * \f[
73  * y_{n+1} = \frac{\left( 2 + \frac{5 h^2}{6} g_n \right) y_n - \left( 1 + \frac{h^2}{12} g_{n-1} \right) y_{n-1}}{\left( 1 + \frac{h^2}{12} g_n \right)}
74  * \f]
75  *
76  * with \f$ y_n = y(x_n) \f$ and \f$ g_n = g(x_n) \f$. Clearly this equation
77  * needs two initial conditions \f$ y_0 \f$ and \f$ y_1 \f$. We integrate the
78  * equation from outside to inside. Since the wavefunction hs to decay to zero
79  * at infinity we choose \f$ y_0 = 0 \f$. Now depending on the number of knots
80  * of wavefunction we decide whether to set \f$ y_1 = \pm\varepsilon \f$.
81  */
82 class Numerov {
83  QuantumDefect const& qd;
84  eigen_dense_double_t xy;
85 public:
86  /** \brief Integration step size */
87  static constexpr double const dx = 0.01;
88 
89  /** \brief Constructor
90  *
91  * The constructor initializes the \p qd member and sets the initial
92  * condition for the integrator. It determines the outer and inner
93  * integration bounds by semiclassical and empirical arguments.
94  *
95  * \param[in] qd Quantum defect data (parameters for the potentials)
96  */
97  Numerov(QuantumDefect const& qd);

99  /** \brief Perform the integration

101  * This performs the integration using %Numerov's method.
102  *
103  * \returns vector with wavefunction amplitude
104  */
105  eigen_dense_double_t integrate();

114  * \returns power kernel
115  */
116  constexpr static inline int power_kernel(int power)

216 /** \brief Find and return index
217  *
218  * Find a value in an Eigen matrix and return its index in the
219  * container to the caller. This is intended to work with
220  * Eigen::Vector or columns of Eigen::Matrix.
221  *
222  * \param[in] x An Eigen matrix type
223  * \param[in] d value to find
224  * \returns index of d
225  * \throws std::runtime_error if the value can't be found
226  */
227 template < typename T >
228 int findidx(T const& x, typename T::Scalar const& d) {
229  int L = 0;
230  int R = x.rows() - 1;
231  for (;;) {
232  if (L > R) throw std::runtime_error("Search failed");
233  int m = (L+R)/2;
234  if (x(m) < d) L = m + 1;
235  if (x(m) > d) R = m - 1;
236  if (x(m) == d) return m;
237  }
238 }

241 /** \brief Compute radial matrix elements
242  *
243  * The radial matrix element can be calculated from the integral
244  * \f[
245  * \langle nlj| \hat{p}^{\mathrm{rad}}_\kappa |n'l'j' \rangle =
246  * \int \Psi^{\text{rad}}_{nlj}(r) \Psi^{\text{rad}}_{n'l'j'}(r) r^{2+\kappa} \; dr
247  * \f]
248  * The part \f$ r^{2+\kappa} \f$ varies with rescaling the domain and is this
249  * given by the power_kernel function.
250  *
251  * \tparam T Method for calculating the wavefunction
252  * \param[in] qd1 Quantum defect data for first atom
253  * \param[in] power Exponent kappa
254  * \param[in] qd2 Quantum defect data for second atom
255  * \returns Radial matrix element
256  */
257 template < typename T >
258 double IntegrateRadialElement(QuantumDefect const& qd1, int power, QuantumDefect const& qd2)
259 {
260  T N1(qd1);
261  T N2(qd2);
262 
263  auto const& xy1 = N1.integrate();
264  auto const& xy2 = N2.integrate();
265  auto const dx = N1.dx;
266 
267  auto const xmin = xy1(0,0) >= xy2(0,0) ? xy1(0,0) : xy2(0,0);
268  auto const xmax = xy1(xy1.rows()-1,0) <= xy2(xy2.rows()-1,0) ? xy1(xy1.rows()-1,0) : xy2(xy2.rows()-1,0);
269 
270  double mu = 0;
271  // If there is an overlap, calculate the matrix element
272  if (xmin <= xmax) {
273  int start1 = findidx(xy1.col(0), xmin);
274  int end1 = findidx(xy1.col(0), xmax);
275  int start2 = findidx(xy2.col(0), xmin);
276  int end2 = findidx(xy2.col(0), xmax);
277 
278  int i1, i2;
279  for (i1 = start1, i2 = start2; i1 < end1 && i2 < end2; ++i1, ++i2)
280  {
281  mu += xy1(i1,1)*xy2(i2,1) * std::pow(xy1(i1,0), T::power_kernel(power)) * dx;
282  }
283  mu = 2*mu;
284  }
285 
286  // If the power is non-zero, the radial matrix element has a unit
287  // that needs to be converted (if the power is zero, unit-free
288  // overlap matrix element is returned)
289  double converter = (power == 0) ? 1 : au2GHz / au2Vcm * std::pow(au2um, power-1);
290 
291  return mu * converter;
292 }
