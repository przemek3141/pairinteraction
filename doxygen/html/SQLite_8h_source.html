73 /** \brief SQLite Statement handle
74  *
75  * This object wraps an SQLite statement. It is iterable in a while
76  * loop with the step() function or in a range-based for loop.
77  */
78 class statement
79 {
80  sqlite3 *m_db;

82  std::string m_sql;

84  bool m_valid;

86  void handle_error(int err)
87  {
88  if (err)
89  throw error(err, sqlite3_errstr(err));

92 public:
93  /** \brief Constructor

95  * This is called in the query function of the the handle object.

103  * This is called in the query function of the the handle object.

106  * \param[in] db unmanaged raw pointer to the sqlite database
107  * \param[in] sql an initial query as a string
108  */

115  /** \brief Set the query string
116  *
117  * \param[in] sql the query string

129  /** \brief Prepare the statement
130  *
131  * Prepare the statement for stepping.
132  *
133  * \throws sqlite::error
134  */
135  void prepare()
136  {
137  sqlite3_stmt *pStmt; // is managed below
138  auto err = sqlite3_prepare_v2(m_db, m_sql.c_str(), -1, &pStmt, nullptr);
139  m_stmt.reset(pStmt);
140 
141  handle_error(err);
142 
143  m_prepared = true;
144  }
145 
146  /** \brief Step the statement
147  *
148  * This steps the statement.
149  *
150  * \returns true if there is a row, false otherwise
151  *
152  * \throws sqlite::error
153  */
154  bool step()
155  {
156  if (!m_prepared)
157  handle_error(SQLITE_MISUSE);
158  if (!m_valid)
159  handle_error(SQLITE_DONE);
160 
161  auto err = sqlite3_step(m_stmt.get());
162 
163  switch (err) {
164  case SQLITE_DONE:
165  m_valid = false;
166  return false;
167  case SQLITE_ROW:
168  m_valid = true;
169  return true;
170  }
171 
172  m_valid = false;
173  handle_error(err);
174  return false;
175  }
176 
177  /** \brief Reset the statement
178  *
179  * Reset the statement so you can rebind values.
180  *
181  * \throws sqlite::error
182  */
183  void reset()
184  {
185  handle_error(sqlite3_reset(m_stmt.get()));
186  m_valid = true;
187  }

189  /** \brief Execute SQLite statements
190  *
191  * The SQL statements are passed to this function as a string and
192  * are executed in-place.
193  *
194  * \param[in] sql SQL statements
195  * \throws sqlite::error
196  */
197  void exec(std::string const &sql)
198  {
199  set(sql);
200  auto err = sqlite3_exec(m_db, m_sql.c_str(), nullptr, nullptr, nullptr);

202  }

272  /** \brief Statement iterator
273  *
274  * The iterator builds upon Boost Iterator Facade to implement an
275  * iterator over the statement with minimal boilerplate. We refer
276  * to the official documentation for the internal functions:
277  * http://www.boost.org/libs/iterator/
278  */
279  class iterator : public boost::iterator_facade<iterator, statement,
280  boost::forward_traversal_tag>
281  {
282  private:
283  friend class boost::iterator_core_access;
284  statement *m_stmt;
285  bool m_done;
286  bool m_end;
287 
288  void increment()
289  {
290  if (!m_done)
291  m_done = m_stmt->step();
292  else
293  m_end = true;
294  }
295 
296  bool equal(iterator const &) const { return m_end; }
297 
298  statement &dereference() const
299  {
300 #ifndef NDEBUG
301  if (m_end)
302  throw std::out_of_range("iterator out of range");
303 #endif
304  return *m_stmt;
305  }
306 
307  public:
308  iterator() : m_stmt{nullptr}, m_done{true}, m_end{true} {}
309 
310  explicit iterator(statement *p) : m_stmt{p}, m_done{false}, m_end{false}
311  {
312  m_stmt->step();
313  }
314  };

317  *
318  * With this, a statement can be iterated in a C++11 range-based for loop.
319  *
320  * \warning This is *not* zero overhead! If you care about
321  * performance use a while loop using step() instead.

326  *

343 /** \brief SQLite Database handle
344  *
345  * This object handles the connection to the underlying SQLite databse. It
346  * provides a high-level object oriented RAII interface to the C-bindings of
347  * SQLite3.
348  */
349 class handle final
350 {
351  std::unique_ptr<sqlite3, decltype(&sqlite3_close)> m_db;
352  int m_threshold;
353 
354  static int busy_handler(void *self, int num_prior_calls)
355  {
356  // Sleep if handler has been called less than num_prior_calls
357  if (num_prior_calls < static_cast<handle *>(self)->m_threshold) {
358  std::this_thread::sleep_for(
359  std::chrono::microseconds(utils::randint(2000, 20000)));
360  return 1;
361  }
362 
363  return 0; // Make sqlite3 return SQLITE_BUSY
364  }
365 
366 public:
367  /** \brief Conversion operator
368  *
369  * The handle object implicitly behaves like the underlying
370  * pointer.
371  *
372  * \returns the raw database pointer
373  */
374  operator sqlite3 *() { return m_db.get(); }
375 
376  /** \brief Constructor
377  *
378  * It takes a filename and a bitmask of parameters to open the
379  * database connection. This is useful if you have to open the
380  * connection, e.g., read-only. By default it uses the standard
381  * flags for opening a database connection. See
382  * https://www.sqlite.org/c3ref/open.html for details.
383  *
384  * \param[in] filename fully-qualified filename of the database
385  * \param[in] flags options to open the database (optional)
386  * \throws sqlite::error
387  */
388  explicit handle(std::string const &filename,
389  int flags = SQLITE_OPEN_READWRITE | SQLITE_OPEN_CREATE)
390  : m_db{nullptr, sqlite3_close}, m_threshold{100000}
391  {
392  sqlite3 *tmp_db;
393  auto err = sqlite3_open_v2(filename.c_str(), &tmp_db, flags, nullptr);
394  m_db.reset(tmp_db);
395 
396  if (err)
397  throw error(err, sqlite3_errmsg(m_db.get()));
398 
399  err = sqlite3_busy_handler(m_db.get(), busy_handler, this);
400 
401  if (err)
402  throw error(err, sqlite3_errmsg(m_db.get()));
403  }
404 
405  /** \brief Execute SQLite statements
406  *
407  * The SQL statements are passed to this function as a string and
408  * are executed in-place.
409  *
410  * \deprecated This is deprecated and should not be used anymore.
411  * The usage of prepared statements in encouraged instead. Calls
412  * to this function are equivalent to and can hence be replaced by
413  * \code
414  * sqlite::statement stmt(db);
415  * stmt.exec(sql);
416  * \endcode
417  *
418  * \param[in] sql SQL statements
419  */
420 #if defined(_MSC_VER)
421  __declspec(deprecated)
422 #else
423  __attribute__((deprecated))
424 #endif
425  void exec(std::string const &sql)
426  {
427  auto err = sqlite3_exec(*this, sql.c_str(), nullptr, nullptr, nullptr);
428  if (err)
429  throw error(err, sqlite3_errmsg(m_db.get()));
430  }
431 };
