308  rows_with_maxval.reserve(overlap_sqrt.cols());

317  maxval = triple.value();

319  }

322  }

968

974  sqnorm += std::pow(std::abs(triple.value()),2);

989  std::vector<double> sqnorm_list(coefficients.rows(),0);
