89  /** \brief Perform sanity checks for convert
90  *
91  * To ensure that dimensions and types are consistent we
92  * introduce this check function.
93  *
94  * \param[in] len Length of the memory
95  * \param[in] nd Number of dimensions
96  * \param[in] dims Initializer list with lengths of dimensions
97  *
98  * \return The total length of the array
99  */
100  template < typename T >
101  void check_array_sanity(int len, int nd, std::initializer_list<T> dims)
102  {
103  if ( len < 1 )
104  throw std::out_of_range(
105  "Trying to create a numpy array with zero or negative element count!");
106 
107  if ( nd != static_cast<int>(dims.size()) )
108  throw std::out_of_range(
109  "Dimension mismatch!");
110 
111  if ( len > std::accumulate( dims.begin(), dims.end(), int{1}, [](int a, int b) { return a*b; } ) )
112  throw std::out_of_range(
113  "Requested dimension is larger than data!");
114  }

116  /** \brief Create a Numpy view of an iterator (implementation)
117  *
118  * This creates a view of the data, i.e. the data still
119  * belongs to the C++ end. When the data is deallocated on
120  * the C++ end trying to access it in Python will probably
121  * cause a segfault. In any case it is undefined behavior.
122  *
123  * \param[in] nd Number of dimensions
124  * \param[in] dim Dimensions of the array
125  * \param[in] dtype Value type of array elements
126  * \param[in] data Pointer to array data
127  *
128  * \return PyObject* containing a Numpy array
129  */
130  template < view_or_copy v, bool const_tag, typename value_type >
131  typename std::enable_if < v == numpy::view, PyObject * >::type
132  convert_impl(int nd, npy_intp * dim, int dtype, void * data, int)
133  {
134  PyObject * ndarray = PyArray_New(&PyArray_Type, nd, dim, dtype, nullptr,
135  data, 0, NPY_ARRAY_FARRAY, nullptr);
136 
137  if ( const_tag )
138  PyArray_CLEARFLAGS(reinterpret_cast<PyArrayObject*>(ndarray),
139  NPY_ARRAY_WRITEABLE);
140  return ndarray;
141  }

143  /** \brief Create a Numpy copy of an iterator (implementation)
144  *
145  * This creates a copy of the data, i.e. the data belongs to
146  * the Python end. The array is marked with NPY_OWNDATA which
147  * should tell the garbage collector to release this memory.
148  *
149  * \param[in] nd Number of dimensions
150  * \param[in] dim Dimensions of the array
151  * \param[in] dtype Value type of array elements
152  * \param[in] data Pointer to array data
153  * \param[in] len Length of data
154  *
155  * \return PyObject* containing a Numpy array
156  */
157  template < view_or_copy v, bool const_tag, typename value_type >
158  typename std::enable_if < v == numpy::copy, PyObject * >::type
159  convert_impl(int nd, npy_intp * dim, int dtype, void * data, int len)
160  {
161  PyObject * ndarray = PyArray_New(&PyArray_Type, nd, dim, dtype, nullptr,
162  nullptr, 0, NPY_ARRAY_FARRAY, nullptr);
163 
164  std::memcpy(PyArray_DATA(reinterpret_cast<PyArrayObject*>(ndarray)),
165  data, len*sizeof(value_type));
166  PyArray_ENABLEFLAGS(reinterpret_cast<PyArrayObject*>(ndarray),
167  NPY_ARRAY_OWNDATA);
168  return ndarray;
169  }

171  /** \brief Create a Numpy array from an iterator (implementation)
172  *
173  * This is the implementation for the conversion of an
174  * arbitrary pointer between \p begin and \p end .
175  *
176  * \param[in] begin Random access iterator pointing to the start
177  * \param[in] end Random access iterator pointing to the end
178  * \param[in] nd Number of dimensions
179  * \param[in] dims Initializer list with lengths of dimensions
180  *
181  * \return PyObject* containing a Numpy array
182  */
183  template < view_or_copy v, typename RAIter, typename T >
184  PyObject * convert(RAIter begin, RAIter end, int nd, std::initializer_list<T> dims,
185  std::random_access_iterator_tag)
186  {
187  using value_type = typename std::iterator_traits<RAIter>::value_type;
188  constexpr auto const_tag = traits::is_pointer_to_const < decltype(&(*begin)) >::value;
189 
190  int const len = std::distance(begin, end);
191 
192  check_array_sanity(len, nd, dims);
193 
194  npy_intp * dim = const_cast <
195  typename traits::pointer_remove_const<decltype(&(*dims.begin()))>::type
196  > ( &(*dims.begin()) );
197  auto dtype = py_type<value_type>::type;
198  void * data = const_cast <
199  typename traits::pointer_remove_const<decltype(&(*begin))>::type
200  > ( &(*begin) );
201 
202  return convert_impl<v, const_tag, value_type>(nd, dim, dtype, data, len);
203  }
204  }

292  /** \brief Create a Numpy array from an Eigen::SparseMatrix
293  *
294  * This is the implementation for the conversion of an
295  * Eigen::SparseMatrix to a scipy.sparse.csc_matrix. We load
296  * the scipy.sparse module and call the constructor for the
297  * csc_matrix which we populate with numpy arrays.
298  *
299  * \param[in] sm T&&
300  *
301  * \return PyObject* containing a Scipy csc_matrix
302  */
303  template < view_or_copy v, typename T >
304  PyObject * sparse_impl(T&& sm)
305  {
306  if ( ! sm.isCompressed() )
307  // we cannot call makeCompressed here because sm might be const
308  throw std::runtime_error("Sparse matrix is not compressed!");
309 
310  numpy::array indptr = numpy::convert<v>(
311  sm.outerIndexPtr(), sm.outerIndexPtr() + sm.outerSize() + 1);
312 
313  numpy::array indices = numpy::convert<v>(
314  sm.innerIndexPtr(), sm.innerIndexPtr() + sm.nonZeros());
315 
316  numpy::array data = numpy::convert<v>(
317  sm.valuePtr(), sm.valuePtr() + sm.nonZeros());
318 
319  int num_rows = sm.rows();
320  int num_cols = sm.cols();
321 
322  char object[] = "csc_matrix";
323  char arglist[] = "(OOO)(ii)";
324  PyObject * scipy = PyImport_ImportModule("scipy.sparse");
325  PyObject * mat = PyObject_CallMethod(
326  scipy, object, arglist, data, indices, indptr, num_rows, num_cols);
327  Py_DECREF(scipy);
328  return mat;
329  }

356  /** \brief Check if a Numpy has given storage order and alignment
357  *
358  * To be convertible to an Eigen matrix type the memory of a
359  * Numpy array has to be FORTRAN style contiguous and has to
360  * be aligned.
361  *
362  * \param[in] ndarray Numpy array
363  */
364  void check_order_and_alignment(PyArrayObject * ndarray)
365  {
366  int const flags = PyArray_FLAGS(ndarray);
367  if ( (flags & NPY_ARRAY_F_CONTIGUOUS) == 0 )
368  throw std::invalid_argument(
369  "The argument is not contiguous or has wrong storage order!");
370  if ( (flags & NPY_ARRAY_ALIGNED) == 0 )
371  throw std::invalid_argument(
372  "The argument is not not aligned!");
373  }

375  /** \brief Deconstruct a numpy array for conversion to dense Eigen type
376  *
377  * \param[in] ndarray_ Numpy array
378  *
379  * \return A tuple of the number of dimensions, an array of
380  * dimensions, and a pointer to the data.
381  */
382  template < typename T, typename Scalar = typename T::Scalar >
383  std::tuple<int, npy_intp *, Scalar *> as_dense_impl(numpy::array ndarray_)
384  {
385  if ( ! ndarray_ || ! PyArray_Check(ndarray_) )
386  throw std::invalid_argument(
387  "The argument is not a valid Numpy array!");
388 
389  PyArrayObject* ndarray = reinterpret_cast<PyArrayObject*>(ndarray_);
390  internal::check_order_and_alignment(ndarray);
391  int const nd = PyArray_NDIM(ndarray);
392  npy_intp * dims = PyArray_SHAPE(ndarray);
393  Scalar * data = reinterpret_cast<Scalar *>(PyArray_DATA(ndarray));
394 
395  if ( PyArray_TYPE(ndarray) != internal::py_type<Scalar>::type )
396  throw std::invalid_argument("Type mismatch.");
397 
398  return std::make_tuple(nd, dims, data);
399  }

403  /** \brief Convert a Numpy array to an Eigen::Matrix
404  *
405  * \warning Experimental! For now the returned Eigen::Map is

408  * \param[in] ndarray Numpy array

410  * \return An Eigen::Map of the Numpy array memory

412 #ifndef SCANNED_BY_DOXYGEN
413  template < typename T, typename Scalar = typename T::Scalar >
414  typename std::enable_if<
415  std::is_same<
416  T,
417  Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic, T::Options,
418  T::MaxRowsAtCompileTime, T::MaxColsAtCompileTime>

421  >::type

472  /** \brief Convert a Numpy array to an Eigen::SparseMatrix
473  *
474  * \warning Experimental! For now the returned Eigen::Map is
475  * read-only.
476  *
477  * \param[in] ndarray Numpy array
478  *
479  * \return An Eigen::Map of the Numpy array memory
480  */
481 #ifndef SCANNED_BY_DOXYGEN
482  template < typename T, typename Scalar = typename T::Scalar,
483  typename StorageIndex = typename T::StorageIndex >
484  typename std::enable_if<
485  std::is_same<
486  T,
487  Eigen::SparseMatrix<Scalar, T::Options, StorageIndex>
488  >::value,
489  Eigen::Map < T const >
490  >::type
491 #else
492  template < typename SparseMatrixType >
493  Eigen::Map < SparseMatrixType >
494 #endif
495  as(numpy::array ndarray)
496  {
497  PyObject * scipy = PyImport_ImportModule("scipy.sparse");
498  PyObject * csc_matrix = PyObject_GetAttrString(scipy, "csc_matrix");
499  if ( ! ndarray || ! PyObject_IsInstance(ndarray, csc_matrix) )
500  throw std::invalid_argument(
501  "The argument is not a valid csc_matrix!");
502  Py_DECREF(csc_matrix);
503  Py_DECREF(scipy);
504 
505  PyObject * tmp;
506 
507  tmp = PyObject_GetAttrString(ndarray, "data");
508  PyArrayObject * data_ = reinterpret_cast<PyArrayObject*>(tmp);
509  Py_DECREF(tmp); // Probably unsafe but otherwise the passed object can never be garbage collected
510  internal::check_order_and_alignment(data_);
511 
512  tmp = PyObject_GetAttrString(ndarray, "indices");
513  PyArrayObject * indices_ = reinterpret_cast<PyArrayObject*>(tmp);
514  Py_DECREF(tmp); // Probably unsafe but otherwise the passed object can never be garbage collected
515  internal::check_order_and_alignment(indices_);
516 
517  tmp = PyObject_GetAttrString(ndarray, "indptr");
518  PyArrayObject * indptr_ = reinterpret_cast<PyArrayObject*>(tmp);
519  Py_DECREF(tmp); // Probably unsafe but otherwise the passed object can never be garbage collected
520  internal::check_order_and_alignment(indptr_);
521 
522  PyObject * shape = PyObject_GetAttrString(ndarray, "shape");
523  int rows, cols;
524  PyArg_ParseTuple(shape, "ii", &rows, &cols);
525  Py_DECREF(shape);
526 
527  npy_intp * nnz = PyArray_SHAPE(data_);
528 
529  Scalar * data = reinterpret_cast<Scalar *>(PyArray_DATA(data_ ));
530  StorageIndex * indices = reinterpret_cast<StorageIndex *>(PyArray_DATA(indices_));
531  StorageIndex * indptr = reinterpret_cast<StorageIndex *>(PyArray_DATA(indptr_ ));
532 
533  if ( PyArray_TYPE(data_) != internal::py_type<Scalar>::type )
534  throw std::invalid_argument("Type mismatch.");
535
