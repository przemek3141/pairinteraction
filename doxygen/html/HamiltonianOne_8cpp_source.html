31 void HamiltonianOne::changeToSpherical(double val_x, double val_y, double val_z, double& val_p, double& val_m, double& val_0) {
32  if(val_y != 0) {
33  std::string msg("For fields with non-zero y-coordinates, a complex data type is needed.");
34  auto context = zmq::context();
35  auto publisher = context.socket(ZMQ_PUB);
36  publisher.connect(zmq::endpoint::name.c_str());
37  publisher.send(">>ERR%s", msg);
38  throw std::runtime_error(msg);
39  }
40  val_p = -val_x/std::sqrt(2);
41  val_m = val_x/std::sqrt(2);
42  val_0 = val_z;
43 }

45 void HamiltonianOne::changeToSpherical(double val_x, double val_y, double val_z, std::complex<double>& val_p, std::complex<double>& val_m, std::complex<double>& val_0) {

48  val_0 = std::complex<double>(val_z,0);

51 void HamiltonianOne::configure(const Configuration &config) {
52  basicconf = basis->getConf();

54  basicconf["diamagnetism"] << config["diamagnetism"];
55 
56  basicconf["deltaESingle"] >> deltaE;
57  basicconf["species1"] >> species;
58 
59  diamagnetism = basicconf["diamagnetism"].str() == "true";
60 
61  config["minBx"] >> min_B_x;
62  config["minBy"] >> min_B_y;
63  config["minBz"] >> min_B_z;
64  config["minEx"] >> min_E_x;
65  config["minEy"] >> min_E_y;
66  config["minEz"] >> min_E_z;
67  config["maxBx"] >> max_B_x;
68  config["maxBy"] >> max_B_y;
69  config["maxBz"] >> max_B_z;
70  config["maxEx"] >> max_E_x;
71  config["maxEy"] >> max_E_y;
72  config["maxEz"] >> max_E_z;
73 
74  if ((min_B_x == max_B_x) &&
75  (min_B_y == max_B_y) &&
76  (min_B_z == max_B_z) &&
77  (min_E_x == max_E_x) &&
78  (min_E_y == max_E_y) &&
79  (min_E_z == max_E_z)) {
80  nSteps = 1;
81  } else {
82  config["steps"] >> nSteps;
83  }
