30 double V(QuantumDefect const& qd, double x) {
31  double Z_l = 1 + (qd.Z - 1)*std::exp(-qd.a1*x)
32  - x*(qd.a3 + qd.a4*x)*std::exp(-qd.a2*x);
33  double V_c = -Z_l / x;
34  double V_p = -qd.ac/(2.*x*x*x*x) * (1-std::exp(-std::pow(x/qd.rc,6)));
35  double V_so = 0.0;
36  if ( qd.l < 4 ) {
37  double alpha = 7.2973525664e-3;// ~1/137 fine-structure constant CODATA 2014
38  V_so = alpha*alpha/(4 * x*x*x) * (qd.j*(qd.j+1) - qd.l*(qd.l+1) - 0.5*(1+0.5));
39  }
40  return V_c + V_p + V_so;
41 }

73 eigen_dense_double_t Numerov::integrate()
74 {
75  using model_potential::g;
76 
77  int const nsteps = xy.rows();
78 
79  // Set the initial condition
80  if ( (qd.n-qd.l) % 2 == 0 )
81  xy(nsteps-2,1) = -1e-10;
82  else
83  xy(nsteps-2,1) = 1e-10;
84 
85  // Perform the integration using Numerov's scheme
86  for (int i = nsteps-3; i >= 0; --i)
87  {
88  double A = (2. + 5./6. * dx*dx * g(qd,xy(i+1,0)*xy(i+1,0))) * xy(i+1,1);
89  double B = (1. - 1./12.* dx*dx * g(qd,xy(i+2,0)*xy(i+2,0))) * xy(i+2,1);
90  double C = 1. - 1./12.* dx*dx * g(qd,xy(i,0)*xy(i,0));
91  xy(i,1) = (A - B)/C;
92  }
93 
94  // Normalization
95  double norm = 0;
96  for (int i = 0; i < nsteps; ++i)
97  norm += xy(i,1)*xy(i,1) * xy(i,0)*xy(i,0) * dx;
98  norm = std::sqrt(2*norm);
99 
100  if ( norm > 0.0 ) {
101  for (int i = 0; i < nsteps; ++i)
102  xy(i,1) /= norm;
103  }
104 
105  return xy;
106 }
