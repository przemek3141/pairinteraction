100  template<class T>
101  void addRotated(const StateOne &state, const size_t &idx, std::vector<Eigen::Triplet<T>> &triplets, WignerD &wigner, const double &alpha, const double &beta, const double &gamma) {
102  // Check whether the angles are compatible to the used data type
103  double tolerance = 1e-16;
104  if (std::is_same<T, double>::value && std::abs(std::remainder(alpha,2*M_PI)) > tolerance) throw std::runtime_error( "If the Euler angle alpha is not a multiple of 2 pi, the Wigner D matrix element is complex and cannot be converted to double." );
105  if (std::is_same<T, double>::value && std::abs(std::remainder(gamma,2*M_PI)) > tolerance) throw std::runtime_error( "If the Euler angle gamma is not a multiple of 2 pi, the Wigner D matrix element is complex and cannot be converted to double." );
106 
107  // Add rotated triplet entries
108  StateOne newstate = state;
109 
110  for (float m = -state.j; m <= state.j; ++m) {
111  newstate.m = m;
112  auto state_iter = states.get<1>().find(newstate);
113 
114  if (state_iter != states.get<1>().end()) {
115  T val = convert<T>(wigner(state.j, state.m, m, alpha, beta, gamma));
116  triplets.push_back(Eigen::Triplet<T>(state_iter->idx, idx, val));
117  } else {
118  std::cerr << "Warning: Incomplete rotation because the basis is lacking some Zeeman levels." << std::endl;
119  }
120  }
121  }
