72 void SystemTwo::setAngle(double a) {
73  if (a != 0 && ordermax > 3) throw std::runtime_error( "A non-zero interaction angle can be directly used only for dipole-dipole interaction.");
74 
75  this->onParameterChange();
76  angle = a;
77 
78  angle_terms[0] = -1.;
79  angle_terms[1] = 1.-3.*std::pow(std::cos(angle),2);
80  angle_terms[2] = -1.5*std::pow(std::sin(angle),2);
81  angle_terms[3] = -3./std::sqrt(2)*std::sin(angle)*std::cos(angle);
82 }

119 void SystemTwo::initializeBasis()
120 {
121  ////////////////////////////////////////////////////////////////////
122  /// Restrict one atom states to the allowed quantum numbers ////////
123  ////////////////////////////////////////////////////////////////////
124 
125  system1.diagonalize(); // it is important to call this method here!
126  system1.restrictN(range_n);
127  system1.restrictL(range_l);
128  system1.restrictJ(range_j);
129  system1.restrictM(range_m);
130  system2.diagonalize(); // it is important to call this method here!
131  system2.restrictN(range_n);
132  system2.restrictL(range_l);
133  system2.restrictJ(range_j);
134  system2.restrictM(range_m);
135 
136  ////////////////////////////////////////////////////////////////////
137  /// Check wther the single atom states fit to the symmetries ///////
138  ////////////////////////////////////////////////////////////////////
139 
140  if (sym_permutation != NA || sym_permutation != NA) {
141  // TODO check system1 == system2
142  }
143 
144  ////////////////////////////////////////////////////////////////////
145  /// Combine one atom states ////////////////////////////////////////
146  ////////////////////////////////////////////////////////////////////
147 
148  // TODO consider further symmetries and check whether they are applicable
149 
150  std::vector<eigen_triplet_t> hamiltonianmatrix_triplets;
151  hamiltonianmatrix_triplets.reserve(system1.getNumVectors()*system2.getNumVectors());
152  states.reserve(system1.getNumStates()*system2.getNumStates());
153  std::vector<eigen_triplet_t> coefficients_triplets; // TODO reserve
154  std::vector<double> sqnorm_list(system1.getNumStates()*system2.getNumStates(), 0);
155 
156  size_t col_new = 0;
157  for (size_t col_1=0; col_1<system1.getNumVectors(); ++col_1) {
158  for (size_t col_2=0; col_2<system2.getNumVectors(); ++col_2) {
159 
160  // In case of inversion symmetry: skip half of the basis vector pairs
161  if ((sym_inversion == EVEN && col_1 <= col_2) || // gerade
162  (sym_inversion == ODD && col_1 < col_2)) { // ungerade
163  continue;
164  }
165 
166  // In case of permutation symmetry: skip half of the basis vector pairs
167  if ((sym_permutation == EVEN && col_1 <= col_2) || // sym
168  (sym_permutation == ODD && col_1 < col_2)) { // asym
169  continue;
170  }
171 
172  // Continue if the pair statet energy is not valid
173  double energy = this->real(system1.getHamiltonianmatrix().coeff(col_1, col_1)+system2.getHamiltonianmatrix().coeff(col_2, col_2));
174  if (!checkIsEnergyValid(energy)) continue;
175 
176  // Store the pair state energy
177  hamiltonianmatrix_triplets.push_back(eigen_triplet_t(col_new, col_new, energy));
178 
179  // Build the basis vector that corresponds to the stored pair state energy
180  for (eigen_iterator_t triple_1(system1.getCoefficients(),col_1); triple_1; ++triple_1) {
181  size_t row_1 = triple_1.row();
182  StateOne state_1 = system1.getStates()[row_1]; // TODO cache states before
183 
184  for (eigen_iterator_t triple_2(system2.getCoefficients(),col_2); triple_2; ++triple_2) {
185  size_t row_2 = triple_2.row();
186  StateOne state_2 = system2.getStates()[row_2]; // TODO cache states before
187 
188  scalar_t value_new = triple_1.value() * triple_2.value();
189 
190  int M = state_1.m+state_2.m;
191  int parityL = std::pow(-1, state_1.l + state_2.l);
192  int parityJ = std::pow(-1, state_1.j + state_2.j);
193  int parityM = std::pow(-1, M);
194 
195  // Consider rotation symmetry
196  if (sym_rotation.count(ARB) == 0 && sym_rotation.count(M) == 0 ) {
197  continue;
198  }
199 
200  // Combine symmetries
201  bool skip_reflection = false;
202  if (col_1 != col_2) {
203  // In case of inversion and permutation symmetry: the inversion symmetric state is already permutation symmetric
204  if (sym_inversion != NA && sym_permutation != NA) {
205  if ( ((sym_inversion == EVEN) ? -parityL : parityL) != ((sym_permutation == EVEN) ? -1 : 1) ) {
206  continue; // the parity under inversion and permutation is different
207  }
208  }
209 
210  // In case of inversion or permutation and reflection symmetry: the inversion or permutation symmetric state is already reflection symmetric
211  if ((sym_inversion != NA || sym_permutation != NA) && sym_reflection != NA && StateTwo({{state_1.element, state_2.element}},{{state_1.n, state_2.n}},{{state_1.l, state_2.l}},{{state_1.j, state_2.j}},{{-state_1.m, -state_2.m}}) == StateTwo(state_2, state_1)) {
212  if (sym_inversion != NA) {
213  if ( ((sym_inversion == EVEN) ? -parityL : parityL) != ((sym_reflection == EVEN) ? parityL*parityJ*parityM : -parityL*parityJ*parityM) ) {
214  continue; // the parity under inversion and reflection is different
215  } else {
216  skip_reflection = true; // the parity under inversion and reflection is the same
217  }
218  } else if (sym_permutation != NA) {
219  if ( ((sym_permutation == EVEN) ? -1 : 1) != ((sym_reflection == EVEN) ? parityL*parityJ*parityM : -parityL*parityJ*parityM) ) {
220  continue; // the parity under permutation and reflection is different
221  } else {
222  skip_reflection = true; // the parity under permutation and reflection is the same
223  }
224  }
225  }
226  }
227 
228  // Adapt the normalization if required by symmetries
229  if (col_1 != col_2) {
230  if (sym_inversion != NA || sym_permutation != NA) {
231  value_new /= std::sqrt(2);
232  }
233  }
234  if (sym_reflection != NA && !skip_reflection) {
235  value_new /= std::sqrt(2)*std::sqrt(2); // the second factor std::sqrt(2) is because of double counting
236  }
237 
238  // Add an entry to the current basis vector
239  this->addCoefficient(StateTwo(state_1, state_2), col_new, value_new, coefficients_triplets, sqnorm_list);
240 
241  // Add further entries to the current basis vector if required by symmetries
242  if (col_1 != col_2) {
243  if (sym_inversion != NA) {
244  scalar_t v = value_new;
245  v *= (sym_inversion == EVEN) ? -parityL : parityL;
246  this->addCoefficient(StateTwo(state_2, state_1), col_new, v, coefficients_triplets, sqnorm_list);
247  } else if (sym_permutation != NA) {
248  scalar_t v = value_new;
249  v *= (sym_permutation == EVEN) ? -1 : 1;
250  this->addCoefficient(StateTwo(state_2, state_1), col_new, v, coefficients_triplets, sqnorm_list);
251  }
252  }
253 
254  if (sym_reflection != NA && !skip_reflection) {
255  scalar_t v = value_new;
256  v *= (sym_reflection == EVEN) ? parityL*parityJ*parityM : -parityL*parityJ*parityM;
257  this->addCoefficient(StateTwo({{state_1.element, state_2.element}},{{state_1.n, state_2.n}},{{state_1.l, state_2.l}},{{state_1.j, state_2.j}},{{-state_1.m, -state_2.m}}), col_new, v, coefficients_triplets, sqnorm_list);
258 
259  if (col_1 != col_2) {
260  if (sym_inversion != NA) {
261  scalar_t v = value_new;
262  v *= (sym_reflection == EVEN) ? parityL*parityJ*parityM : -parityL*parityJ*parityM;
263  v *= (sym_inversion == EVEN) ? -parityL : parityL;
264  this->addCoefficient(StateTwo({{state_2.element, state_1.element}},{{state_2.n, state_1.n}},{{state_2.l, state_1.l}},{{state_2.j, state_1.j}},{{-state_2.m, -state_1.m}}), col_new, v, coefficients_triplets, sqnorm_list);
265  } else if (sym_permutation != NA) {
266  scalar_t v = value_new;
267  v *= (sym_reflection == EVEN) ? parityL*parityJ*parityM : -parityL*parityJ*parityM;
268  v *= (sym_permutation == EVEN) ? -1 : 1;
269  this->addCoefficient(StateTwo({{state_2.element, state_1.element}},{{state_2.n, state_1.n}},{{state_2.l, state_1.l}},{{state_2.j, state_1.j}},{{-state_2.m, -state_1.m}}), col_new, v, coefficients_triplets, sqnorm_list);
270  }
271  }
272  }
273  }
274  }
275 
276  ++col_new;
277  }
278  }
279 
280  // Delete unecessary storage
281  system1 = SystemOne(element[0]);
282  system2 = SystemOne(element[1]);
283 
284  // Build data
285  states.shrink_to_fit();
286 
287  coefficients.resize(states.size(),col_new);
288  coefficients.setFromTriplets(coefficients_triplets.begin(), coefficients_triplets.end());
289  coefficients_triplets.clear();
290 
291  hamiltonianmatrix.resize(col_new,col_new);
292  hamiltonianmatrix.setFromTriplets(hamiltonianmatrix_triplets.begin(), hamiltonianmatrix_triplets.end());
293  hamiltonianmatrix_triplets.clear();
294 
295  ////////////////////////////////////////////////////////////////////
296  /// Remove vectors with too small norm /////////////////////////////
297  ////////////////////////////////////////////////////////////////////
298 
299  // Build transformator and remove vectors (if the squared norm is too small)
300  std::vector<eigen_triplet_t> triplets_transformator;
301  triplets_transformator.reserve(coefficients.cols());
302 
303  size_t idx_new = 0;
304  for (int idx=0; idx<coefficients.cols(); ++idx) { // idx = col = num basis vector
305  double_t sqnorm = 0;
306 
307  // Calculate the square norm of the columns of the coefficient matrix
308  for (eigen_iterator_t triple(coefficients,idx); triple; ++triple) {
309  sqnorm += std::pow(std::abs(triple.value()),2);
310  }
311 
312  if (sqnorm > 0.05) {
313  triplets_transformator.push_back(eigen_triplet_t(idx,idx_new++,1));
314  }
315  }
316 
317  this->applyRightsideTransformator(triplets_transformator);
318 
319  ////////////////////////////////////////////////////////////////////
320  /// Remove states that barely occur ////////////////////////////////
321  ////////////////////////////////////////////////////////////////////
322 
323  // Build transformator and remove states if the squared norm is to small
324  removeRestrictedStates([=](const enumerated_state<StateTwo> &entry) -> bool { return sqnorm_list[entry.idx] > 0.05; } );
325 
326  /*////////////////////////////////////////////////////////////////////
327  /// Sort states ////////////////////////////////////////////////////
328  ////////////////////////////////////////////////////////////////////
329

333  std::vector<size_t> idx_sorted(states.size());
334  iota(idx_sorted.begin(), idx_sorted.end(), 0);

337  std::sort(idx_sorted.begin(), idx_sorted.end(), [=](size_t i1, size_t i2) {return this->states[i1].state < this->states[i2].state;});

360 void SystemTwo::initializeInteraction() {
361  if (distance == std::numeric_limits<double>::max()) return;
362 
363  ////////////////////////////////////////////////////////////////////
364  /// Prepare the calculation of the interaction /////////////////////
365  ////////////////////////////////////////////////////////////////////
366 
367  // Check if something to do
368  double tolerance = 1e-24;
369 
370  std::vector<bool> calculation_required(4, false);
371  std::vector<int> orange;
372 
373  if (angle != 0) { // setAngle and setOrder take care that a non-zero angle cannot occur for other interaction than dipole-dipole
374 
375  for (size_t i = 0; i < 4; ++i) {
376  if (std::abs(angle_terms[i]) > tolerance && interaction_angulardipole.find(i) == interaction_angulardipole.end()) {
377  calculation_required[i] = true;
378  }
379  }
380 
381  } else {
382 
383  for (unsigned int order = 3; order <= ordermax; ++order) {
384  if (interaction_multipole.find(order) == interaction_multipole.end()) {
385  orange.push_back(order);
386  }
387  }
388 
389  }
390 
391  // Precalculate matrix elements
392  std::string matrixelementsdir = "";
393  if (!cachedir.empty()) matrixelementsdir = (cachedir / "cache_elements.db").string(); // TODO do this in the MatrixElements class, just pass cachedir as an argument to the constructor
394 
395  MatrixElements matrixelements1(element[0], matrixelementsdir);
396  MatrixElements matrixelements2(element[1], matrixelementsdir);
397 
398  auto states1 = this->getStatesFirst();
399  auto states2 = this->getStatesSecond();
400 
401  for (unsigned int kappa = 1; kappa <= ordermax-2; ++kappa) {
402  matrixelements1.precalculateMultipole(states1, kappa);
403  matrixelements2.precalculateMultipole(states2, kappa); // TODO check whether system1 == system2
404  }
405 
406  ////////////////////////////////////////////////////////////////////
407  /// Generate the interaction in the canonical basis ////////////////
408  ////////////////////////////////////////////////////////////////////
409 
410  std::unordered_map<int, std::vector<eigen_triplet_t>> interaction_angulardipole_triplets; // TODO reserve
411  std::unordered_map<int, std::vector<eigen_triplet_t>> interaction_multipole_triplets; // TODO reserve
412 
413  // Loop over column entries
414  for (const auto &c: states) { // TODO parallelization
415  if (c.state.element.empty()) continue; // TODO artifical states TODO [dummystates]
416 
417  // Loop over row entries
418  for (const auto &r: states) {
419 
420  if (r.state.element.empty()) continue; // TODO artifical states TODO [dummystates]
421  if (r.idx < c.idx) continue;
422 
423  int q1 = r.state.first().m-c.state.first().m;
424  int q2 = r.state.second().m-c.state.second().m;
425 
426  if (angle != 0) { // setAngle and setOrder take care that a non-zero angle cannot occur for other interaction than dipole-dipole
427 
428  // Angular dependent dipole-dipole interaction
429  if (selectionRulesMultipole(r.state.first(), c.state.first(), 1) && selectionRulesMultipole(r.state.second(), c.state.second(), 1)) {
430  if (q1 == 0 && q2 == 0 && calculation_required[1]) {
431  scalar_t val = inverse_electric_constant * matrixelements1.getMultipole(r.state.first(), c.state.first(), 1) *
432  matrixelements2.getMultipole(r.state.second(), c.state.second(), 1);
433 
434  this->addTriplet(interaction_angulardipole_triplets[1], r.idx, c.idx, val);
435 
436  } else if (q1 != 0 && q2 != 0 && q1+q2 == 0 && (calculation_required[0] || calculation_required[2])) {
437  scalar_t val = inverse_electric_constant * matrixelements1.getMultipole(r.state.first(), c.state.first(), 1) *
438  matrixelements2.getMultipole(r.state.second(), c.state.second(), 1);
439 
440  if (calculation_required[0]) this->addTriplet(interaction_angulardipole_triplets[0], r.idx, c.idx, val);
441  if (calculation_required[2]) this->addTriplet(interaction_angulardipole_triplets[2], r.idx, c.idx, -val);
442 
443  } else if (std::abs(q1+q2) == 1 && calculation_required[3]) {
444  scalar_t val = inverse_electric_constant * matrixelements1.getMultipole(r.state.first(), c.state.first(), 1) *
445  matrixelements2.getMultipole(r.state.second(), c.state.second(), 1);
446 
447  if (q1 == 1 || q2 == 1) this->addTriplet(interaction_angulardipole_triplets[3], r.idx, c.idx, -val);
448  else this->addTriplet(interaction_angulardipole_triplets[3], r.idx, c.idx, val);
449 
450  } else if (std::abs(q1+q2) == 2 && calculation_required[2]) {
451  scalar_t val = inverse_electric_constant * matrixelements1.getMultipole(r.state.first(), c.state.first(), 1) *
452  matrixelements2.getMultipole(r.state.second(), c.state.second(), 1);
453 
454  this->addTriplet(interaction_angulardipole_triplets[2], r.idx, c.idx, val);
455  }
456  }
457 
458  } else {
459 
460  // Multipole interaction
461  if (q1 + q2 == 0) { // total momentum conserved
462  for (const auto &order : orange) {
463  double val = 0;
464  for (int kappa1 = 1; kappa1 <= order-2; ++kappa1) {
465  int kappa2 = order-1-kappa1;
466  if (selectionRulesMultipole(r.state.first(), c.state.first(), kappa1) && selectionRulesMultipole(r.state.second(), c.state.second(), kappa2)) {
467  double binomials = boost::math::binomial_coefficient<double>(kappa1+kappa2, kappa1+q1)*boost::math::binomial_coefficient<double>(kappa1+kappa2, kappa2-q2);
468  val += inverse_electric_constant * std::pow(-1,kappa2) * std::sqrt(binomials) * matrixelements1.getMultipole(r.state.first(), c.state.first(), kappa1)*
469  matrixelements2.getMultipole(r.state.second(), c.state.second(), kappa2);
470  }
471  }
472 
473  this->addTriplet(interaction_multipole_triplets[order], r.idx, c.idx, val);
474  }
475  }
476 
477  }
478  }
479  }
480 
481  ////////////////////////////////////////////////////////////////////
482  /// Build and transform the interaction to the used basis //////////
483  ////////////////////////////////////////////////////////////////////
484 
485  for (size_t i = 0; i < calculation_required.size(); ++i) {
486  if (!calculation_required[i]) continue;
487  interaction_angulardipole[i].resize(states.size(),states.size());
488  interaction_angulardipole[i].setFromTriplets(interaction_angulardipole_triplets[i].begin(), interaction_angulardipole_triplets[i].end());
489  interaction_angulardipole_triplets[i].clear();
490 
491  interaction_angulardipole[i] = coefficients.adjoint()*interaction_angulardipole[i]*coefficients;
492  }
493 
494  for (const auto &i : orange) {
495  interaction_multipole[i].resize(states.size(),states.size());
496  interaction_multipole[i].setFromTriplets(interaction_multipole_triplets[i].begin(), interaction_multipole_triplets[i].end());
497  interaction_multipole_triplets[i].clear();
498 
499  interaction_multipole[i] = coefficients.adjoint()*interaction_multipole[i]*coefficients;
500  }
501 }

507 void SystemTwo::addInteraction() {
508  if (distance == std::numeric_limits<double>::max()) return;
509 
510  // Build the total Hamiltonian
511  double tolerance = 1e-24;
512

595 void SystemTwo::incorporate(SystemBase<StateTwo> &system) {
596  // Combine parameters
597  if (element[0] != dynamic_cast<SystemTwo&>(system).element[0]) throw std::runtime_error("The value of the variable 'element' must be the same for both systems.");
598  if (element[1] != dynamic_cast<SystemTwo&>(system).element[1]) throw std::runtime_error("The value of the variable 'element' must be the same for both systems.");
599  if (distance != dynamic_cast<SystemTwo&>(system).distance) throw std::runtime_error("The value of the variable 'distance' must be the same for both systems.");
600  if (angle != dynamic_cast<SystemTwo&>(system).angle) throw std::runtime_error("The value of the variable 'angle' must be the same for both systems."); // implies that angle_terms is the same, too
601  if (ordermax != dynamic_cast<SystemTwo&>(system).ordermax) throw std::runtime_error("The value of the variable 'ordermax' must be the same for both systems.");
602 
603  // Combine symmetries
604  unsigned int num_different_symmetries = 0;
605  if (sym_permutation != dynamic_cast<SystemTwo&>(system).sym_permutation) {
606  sym_permutation = NA;
607  ++num_different_symmetries;
608  }
609  if (sym_inversion != dynamic_cast<SystemTwo&>(system).sym_inversion) {
610  sym_inversion = NA;
611  ++num_different_symmetries;
612  }
613  if (sym_reflection != dynamic_cast<SystemTwo&>(system).sym_reflection) {
614  sym_reflection = NA;
615  ++num_different_symmetries;
616  }
617  if (!std::equal(sym_rotation.begin(), sym_rotation.end(), dynamic_cast<SystemTwo&>(system).sym_rotation.begin())) {
618  if (sym_rotation.count(ARB)!=0 || dynamic_cast<SystemTwo&>(system).sym_rotation.count(ARB)!=0) {
619  sym_rotation = {ARB};
620  } else {
621  sym_rotation.insert(dynamic_cast<SystemTwo&>(system).sym_rotation.begin(), dynamic_cast<SystemTwo&>(system).sym_rotation.end());
622  }
623  ++num_different_symmetries;
624  }
625  // if (num_different_symmetries > 1) std::cerr << "Warning: The systems differ in more than one symmetry. For the combined system, the notion of symmetries might be meaningless." << std::endl; // TODO let the user enable/disable this warning
626 
627  // Clear cached interaction
628  this->deleteInteraction();
629 }
