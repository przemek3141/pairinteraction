24 BOOST_AUTO_TEST_CASE(sqlite_query_test)

31 
32  sqlite::statement stmt(db);
33  BOOST_CHECK_NO_THROW(stmt.set("This is not valid SQL"));
34  BOOST_CHECK_THROW(stmt.step(), sqlite::error);
35  BOOST_CHECK_THROW(stmt.prepare(), sqlite::error);
36  BOOST_CHECK_THROW(stmt.exec("Neither is this"), sqlite::error);
37 
38  // Check string calling
39  std::string string_query("create table test(text,integer,real);");
40  BOOST_CHECK_NO_THROW(stmt.reset());
41  BOOST_CHECK_NO_THROW(stmt.set(string_query));
42  BOOST_CHECK_NO_THROW(stmt.prepare());
43  BOOST_CHECK_EQUAL(stmt.step(), false);
44 
45  // Check stringstream calling
46  std::stringstream ss_query;
47  ss_query << "insert into test values(?1,?2,?3);";
48  BOOST_CHECK_NO_THROW(stmt.reset());

50  // Insert some stuff

52  BOOST_CHECK_NO_THROW(stmt.bind(1, "Hello World!"));
53  BOOST_CHECK_NO_THROW(stmt.bind(2, 1729));
54  BOOST_CHECK_NO_THROW(stmt.bind(3, 0.5));
55  BOOST_CHECK_EQUAL(stmt.step(), false);
56  // Reuse the query set above
57  BOOST_CHECK_NO_THROW(stmt.reset());
58  BOOST_CHECK_NO_THROW(stmt.prepare());
59  BOOST_CHECK_NO_THROW(stmt.bind(1, "Goodbye Earth!"));
60  BOOST_CHECK_NO_THROW(stmt.bind(2, 42));
61  BOOST_CHECK_NO_THROW(stmt.bind(3, 1.125));

64
