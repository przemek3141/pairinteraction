251  assert(fvec.size()==n);
252  for (VectorXd::Index k = 0; k < n; k++)
253  {
254  temp = (3. - 2.*x[k])*x[k];
255  temp1 = 0.;

257  temp2 = 0.;
258  if (k != n-1) temp2 = x[k+1];

262  }

269  {
270  for (VectorXd::Index j = 0; j < n; j++)
271  fjac(k,j) = 0.;
272  fjac(k,k) = 3.- 4.*x[k];
273  if (k) fjac(k,k-1) = -1.;
