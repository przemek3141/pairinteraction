41 bool selectionRulesMultipole(StateOne const& state1, StateOne const& state2, int kappa, int q) {
42  bool validL = (abs(state1.l-state2.l) <= kappa) && (kappa%2 == abs(state1.l-state2.l)%2);
43  bool validJ = (fabs(state1.j-state2.j) <= kappa) && (state1.j+state2.j >= kappa);
44  bool validM = state1.m == state2.m+q;
45  bool validQ = abs(q) <= kappa;
46  bool noZero = !(kappa == 2 && state1.j == state2.j && state2.j == 1.5 && state1.m == -state2.m && fabs(state1.m - state2.m) == 1);
47  return validL && validJ && validM && validQ && noZero;
48 }

102 real_t MatrixElements::getMagneticMomentum(StateOne const& state_row, StateOne const& state_col) {
103  real_t val = muB * cache_radial[0][StateTwo({{state_row.n, state_col.n}}, {{state_row.l, state_col.l}}, {{0,0}}, {{state_row.j, state_col.j}}, {{0,0}}).order()] *
104  cache_angular[1][StateTwo({{0, 0}}, {{0, 0}}, {{0,0}}, {{state_row.j, state_col.j}}, {{state_row.m, state_col.m}})] *
